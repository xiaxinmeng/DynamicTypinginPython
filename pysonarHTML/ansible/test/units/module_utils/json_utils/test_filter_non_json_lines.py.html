  23 from units.compat import unittest
  24 from ansible.module_utils.json_utils import _filter_non_json_lines
  25 
  26 
  27 class TestAnsibleModuleExitJson(unittest.TestCase):
  28     single_line_json_dict = u"""{"key": "value", "olá": "mundo"}"""
  29     single_line_json_array = u"""["a","b","c"]"""
  30     multi_line_json_dict = u"""{
  31 "key":"value"
  32 }"""
  33     multi_line_json_array = u"""[
  34 "a",
  35 "b",
  36 "c"]"""
  37 
  38     all_inputs = [
  39         single_line_json_dict,
  40         single_line_json_array,
  41         multi_line_json_dict,
  42         multi_line_json_array
  43     ]
  44 
  45     junk = [u"single line of junk", u"line 1/2 of junk\nline 2/2 of junk"]
  46 
  47     unparsable_cases = (
  48         u'No json here',
  49         u'"olá": "mundo"',
  50         u'{"No json": "ending"',
  51         u'{"wrong": "ending"]',
  52         u'["wrong": "ending"}',
  53     )
  54 
  55     def test_just_json(self):
  56         for i in self.all_inputs:
  57             filtered, warnings = _filter_non_json_lines(i)
  58             self.assertEqual(filtered, i)
  59             self.assertEqual(warnings, [])
  60 
  61     def test_leading_junk(self):
  62         for i in self.all_inputs:
  63             for j in self.junk:
  64                 filtered, warnings = _filter_non_json_lines(j + "\n" + i)
  65                 self.assertEqual(filtered, i)
  66                 self.assertEqual(warnings, [])
  67 
  68     def test_trailing_junk(self):
  69         for i in self.all_inputs:
  70             for j in self.junk:
  71                 filtered, warnings = _filter_non_json_lines(i + "\n" + j)
  72                 self.assertEqual(filtered, i)
  73                 self.assertEqual(warnings, [u"Module invocation had junk after the JSON data: %s" % j.strip()])
  74 
  75     def test_leading_and_trailing_junk(self):
  76         for i in self.all_inputs:
  77             for j in self.junk:
  78                 filtered, warnings = _filter_non_json_lines("\n".join([j, i, j]))
  79                 self.assertEqual(filtered, i)
  80                 self.assertEqual(warnings, [u"Module invocation had junk after the JSON data: %s" % j.strip()])
  81 
  82     def test_unparsable_filter_non_json_lines(self):
  83         for i in self.unparsable_cases:
  84             self.assertRaises(
  85                 ValueError,
  86                 _filter_non_json_lines,
  87                 data=i
  88             )
