  74     @abstractmethod
  75     def run(self, tmp=None, task_vars=None):
  76         """ Action Plugins should implement this method to perform their
  77         tasks.  Everything else in this base class is a helper method for the
  78         action plugin to do that.
  79 
  80         :kwarg tmp: Deprecated parameter.  This is no longer used.  An action plugin that calls
  81             another one and wants to use the same remote tmp for both should set
  82             self._connection._shell.tmpdir rather than this parameter.
  83         :kwarg task_vars: The variables (host vars, group vars, config vars,
  84             etc) associated with this task.
  85         :returns: dictionary of results from the module
  86 
  87         Implementors of action modules may find the following variables especially useful:
  88 
  89         * Module parameters.  These are stored in self._task.args
  90         """
  91 
  92         result = {}
  93 
  94         if tmp is not None:
  95             result['warning'] = ['ActionModule.run() no longer honors the tmp parameter. Action'
  96                                  ' plugins should set self._connection._shell.tmpdir to share'
  97                                  ' the tmpdir']
  98         del tmp
  99 
 100         if self._task.async_val and not self._supports_async:
 101             raise AnsibleActionFail('async is not supported for this task.')
 102         elif self._play_context.check_mode and not self._supports_check_mode:
 103             raise AnsibleActionSkip('check mode is not supported for this task.')
 104         elif self._task.async_val and self._play_context.check_mode:
 105             raise AnsibleActionFail('check mode and async cannot be used on same task.')
 106 
 107         # Error if invalid argument is passed
 108         if self._VALID_ARGS:
 109             task_opts = frozenset(self._task.args.keys())
 110             bad_opts = task_opts.difference(self._VALID_ARGS)
 111             if bad_opts:
 112                 raise AnsibleActionFail('Invalid options for %s: %s' % (self._task.action, ','.join(list(bad_opts))))
 113 
 114         if self._connection._shell.tmpdir is None and self._early_needs_tmp_path():
 115             self._make_tmp_path()
 116 
 117         return result

 156     def _configure_module(self, module_name, module_args, task_vars):
 157         '''
 158         Handles the loading and templating of the module code through the
 159         modify_module() function.
 160         '''
 161         if self._task.delegate_to:
 162             use_vars = task_vars.get('ansible_delegated_vars')[self._task.delegate_to]
 163         else:
 164             use_vars = task_vars
 165 
 166         # Search module path(s) for named module.
 167         for mod_type in self._connection.module_implementation_preferences:
 168             # Check to determine if PowerShell modules are supported, and apply
 169             # some fixes (hacks) to module name + args.
 170             if mod_type == '.ps1':
 171                 # FIXME: This should be temporary and moved to an exec subsystem plugin where we can define the mapping
 172                 # for each subsystem.
 173                 win_collection = 'ansible.windows'
 174 
 175                 # async_status, win_stat, win_file, win_copy, and win_ping are not just like their
 176                 # python counterparts but they are compatible enough for our
 177                 # internal usage
 178                 if module_name in ('stat', 'file', 'copy', 'ping') and self._task.action != module_name:
 179                     module_name = '%s.win_%s' % (win_collection, module_name)
 180                 elif module_name in ['async_status']:
 181                     module_name = '%s.%s' % (win_collection, module_name)
 182 
 183                 # Remove extra quotes surrounding path parameters before sending to module.
 184                 if resource_from_fqcr(module_name) in ['win_stat', 'win_file', 'win_copy', 'slurp'] and module_args and \
 185                         hasattr(self._connection._shell, '_unquote'):
 186                     for key in ('src', 'dest', 'path'):
 187                         if key in module_args:
 188                             module_args[key] = self._connection._shell._unquote(module_args[key])
 189 
 190             result = self._shared_loader_obj.module_loader.find_plugin_with_context(module_name, mod_type, collection_list=self._task.collections)
 191 
 192             if not result.resolved:
 193                 if result.redirect_list and len(result.redirect_list) > 1:
 194                     # take the last one in the redirect list, we may have successfully jumped through N other redirects
 195                     target_module_name = result.redirect_list[-1]
 196 
 197                     raise AnsibleError("The module {0} was redirected to {1}, which could not be loaded.".format(module_name, target_module_name))
 198 
 199             module_path = result.plugin_resolved_path
 200             if module_path:
 201                 break
 202         else:  # This is a for-else: http://bit.ly/1ElPkyg
 203             raise AnsibleError("The module %s was not found in configured module paths" % (module_name))
 204 
 205         # insert shared code and arguments into the module
 206         final_environment = dict()
 207         self._compute_environment_string(final_environment)
 208 
 209         become_kwargs = {}
 210         if self._connection.become:
 211             become_kwargs['become'] = True
 212             become_kwargs['become_method'] = self._connection.become.name
 213             become_kwargs['become_user'] = self._connection.become.get_option('become_user',
 214                                                                               playcontext=self._play_context)
 215             become_kwargs['become_password'] = self._connection.become.get_option('become_pass',
 216                                                                                   playcontext=self._play_context)
 217             become_kwargs['become_flags'] = self._connection.become.get_option('become_flags',
 218                                                                                playcontext=self._play_context)
 219 
 220         # modify_module will exit early if interpreter discovery is required; re-run after if necessary
 221         for dummy in (1, 2):
 222             try:
 223                 (module_data, module_style, module_shebang) = modify_module(module_name, module_path, module_args, self._templar,
 224                                                                             task_vars=use_vars,
 225                                                                             module_compression=self._play_context.module_compression,
 226                                                                             async_timeout=self._task.async_val,
 227                                                                             environment=final_environment,
 228                                                                             **become_kwargs)
 229                 break
 230             except InterpreterDiscoveryRequiredError as idre:
 231                 self._discovered_interpreter = AnsibleUnsafeText(discover_interpreter(
 232                     action=self,
 233                     interpreter_name=idre.interpreter_name,
 234                     discovery_mode=idre.discovery_mode,
 235                     task_vars=use_vars))
 236 
 237                 # update the local task_vars with the discovered interpreter (which might be None);
 238                 # we'll propagate back to the controller in the task result
 239                 discovered_key = 'discovered_interpreter_%s' % idre.interpreter_name
 240 
 241                 # update the local vars copy for the retry
 242                 use_vars['ansible_facts'][discovered_key] = self._discovered_interpreter
 243 
 244                 # TODO: this condition prevents 'wrong host' from being updated
 245                 # but in future we would want to be able to update 'delegated host facts'
 246                 # irrespective of task settings
 247                 if not self._task.delegate_to or self._task.delegate_facts:
 248                     # store in local task_vars facts collection for the retry and any other usages in this worker
 249                     task_vars['ansible_facts'][discovered_key] = self._discovered_interpreter
 250                     # preserve this so _execute_module can propagate back to controller as a fact
 251                     self._discovered_interpreter_key = discovered_key
 252                 else:
 253                     task_vars['ansible_delegated_vars'][self._task.delegate_to]['ansible_facts'][discovered_key] = self._discovered_interpreter
 254 
 255         return (module_style, module_shebang, module_data, module_path)

 257     def _compute_environment_string(self, raw_environment_out=None):
 258         '''
 259         Builds the environment string to be used when executing the remote task.
 260         '''
 261 
 262         final_environment = dict()
 263         if self._task.environment is not None:
 264             environments = self._task.environment
 265             if not isinstance(environments, list):
 266                 environments = [environments]
 267 
 268             # The order of environments matters to make sure we merge
 269             # in the parent's values first so those in the block then
 270             # task 'win' in precedence
 271             for environment in environments:
 272                 if environment is None or len(environment) == 0:
 273                     continue
 274                 temp_environment = self._templar.template(environment)
 275                 if not isinstance(temp_environment, dict):
 276                     raise AnsibleError("environment must be a dictionary, received %s (%s)" % (temp_environment, type(temp_environment)))
 277                 # very deliberately using update here instead of combine_vars, as
 278                 # these environment settings should not need to merge sub-dicts
 279                 final_environment.update(temp_environment)
 280 
 281         if len(final_environment) > 0:
 282             final_environment = self._templar.template(final_environment)
 283 
 284         if isinstance(raw_environment_out, dict):
 285             raw_environment_out.clear()
 286             raw_environment_out.update(final_environment)
 287 
 288         return self._connection._shell.env_prefix(**final_environment)

 297     def _is_pipelining_enabled(self, module_style, wrap_async=False):
 298         '''
 299         Determines if we are required and can do pipelining
 300         '''
 301 
 302         try:
 303             is_enabled = self._connection.get_option('pipelining')
 304         except (KeyError, AttributeError, ValueError):
 305             is_enabled = self._play_context.pipelining
 306 
 307         # winrm supports async pipeline
 308         # TODO: make other class property 'has_async_pipelining' to separate cases
 309         always_pipeline = self._connection.always_pipeline_modules
 310 
 311         # su does not work with pipelining
 312         # TODO: add has_pipelining class prop to become plugins
 313         become_exception = (self._connection.become.name if self._connection.become else '') != 'su'
 314 
 315         # any of these require a true
 316         conditions = [
 317             self._connection.has_pipelining,    # connection class supports it
 318             is_enabled or always_pipeline,      # enabled via config or forced via connection (eg winrm)
 319             module_style == "new",              # old style modules do not support pipelining
 320             not C.DEFAULT_KEEP_REMOTE_FILES,    # user wants remote files
 321             not wrap_async or always_pipeline,  # async does not normally support pipelining unless it does (eg winrm)
 322             become_exception,
 323         ]
 324 
 325         return all(conditions)

 335     def _get_remote_user(self):
 336         ''' consistently get the 'remote_user' for the action plugin '''
 337         # TODO: use 'current user running ansible' as fallback when moving away from play_context
 338         # pwd.getpwuid(os.getuid()).pw_name
 339         remote_user = None
 340         try:
 341             remote_user = self._connection.get_option('remote_user')
 342         except KeyError:
 343             # plugin does not have remote_user option, fallback to default and/play_context
 344             remote_user = getattr(self._connection, 'default_user', None) or self._play_context.remote_user
 345         except AttributeError:
 346             # plugin does not use config system, fallback to old play_context
 347             remote_user = self._play_context.remote_user
 348         return remote_user

 350     def _is_become_unprivileged(self):
 351         '''
 352         The user is not the same as the connection user and is not part of the
 353         shell configured admin users
 354         '''
 355         # if we don't use become then we know we aren't switching to a
 356         # different unprivileged user
 357         if not self._connection.become:
 358             return False
 359 
 360         # if we use become and the user is not an admin (or same user) then
 361         # we need to return become_unprivileged as True
 362         admin_users = self._get_admin_users()
 363         remote_user = self._get_remote_user()
 364         become_user = self.get_become_option('become_user')
 365         return bool(become_user and become_user not in admin_users + [remote_user])

 367     def _make_tmp_path(self, remote_user=None):
 368         '''
 369         Create and return a temporary path on a remote box.
 370         '''
 371 
 372         # Network connection plugins (network_cli, netconf, etc.) execute on the controller, rather than the remote host.
 373         # As such, we want to avoid using remote_user for paths  as remote_user may not line up with the local user
 374         # This is a hack and should be solved by more intelligent handling of remote_tmp in 2.7
 375         if getattr(self._connection, '_remote_is_local', False):
 376             tmpdir = C.DEFAULT_LOCAL_TMP
 377         else:
 378             # NOTE: shell plugins should populate this setting anyways, but they dont do remote expansion, which
 379             # we need for 'non posix' systems like cloud-init and solaris
 380             tmpdir = self._remote_expand_user(self.get_shell_option('remote_tmp', default='~/.ansible/tmp'), sudoable=False)
 381 
 382         become_unprivileged = self._is_become_unprivileged()
 383         basefile = self._connection._shell._generate_temp_dir_name()
 384         cmd = self._connection._shell.mkdtemp(basefile=basefile, system=become_unprivileged, tmpdir=tmpdir)
 385         result = self._low_level_execute_command(cmd, sudoable=False)
 386 
 387         # error handling on this seems a little aggressive?
 388         if result['rc'] != 0:
 389             if result['rc'] == 5:
 390                 output = 'Authentication failure.'
 391             elif result['rc'] == 255 and self._connection.transport in ('ssh',):
 392 
 393                 if self._play_context.verbosity > 3:
 394                     output = u'SSH encountered an unknown error. The output was:\n%s%s' % (result['stdout'], result['stderr'])
 395                 else:
 396                     output = (u'SSH encountered an unknown error during the connection. '
 397                               'We recommend you re-run the command using -vvvv, which will enable SSH debugging output to help diagnose the issue')
 398 
 399             elif u'No space left on device' in result['stderr']:
 400                 output = result['stderr']
 401             else:
 402                 output = ('Failed to create temporary directory.'
 403                           'In some cases, you may have been able to authenticate and did not have permissions on the target directory. '
 404                           'Consider changing the remote tmp path in ansible.cfg to a path rooted in "/tmp", for more error information use -vvv. '
 405                           'Failed command was: %s, exited with result %d' % (cmd, result['rc']))
 406             if 'stdout' in result and result['stdout'] != u'':
 407                 output = output + u", stdout output: %s" % result['stdout']
 408             if self._play_context.verbosity > 3 and 'stderr' in result and result['stderr'] != u'':
 409                 output += u", stderr output: %s" % result['stderr']
 410             raise AnsibleConnectionFailure(output)
 411         else:
 412             self._cleanup_remote_tmp = True
 413 
 414         try:
 415             stdout_parts = result['stdout'].strip().split('%s=' % basefile, 1)
 416             rc = self._connection._shell.join_path(stdout_parts[-1], u'').splitlines()[-1]
 417         except IndexError:
 418             # stdout was empty or just space, set to / to trigger error in next if
 419             rc = '/'
 420 
 421         # Catch failure conditions, files should never be
 422         # written to locations in /.
 423         if rc == '/':
 424             raise AnsibleError('failed to resolve remote temporary directory from %s: `%s` returned empty string' % (basefile, cmd))
 425 
 426         self._connection._shell.tmpdir = rc
 427 
 428         return rc

 430     def _should_remove_tmp_path(self, tmp_path):
 431         '''Determine if temporary path should be deleted or kept by user request/config'''
 432         return tmp_path and self._cleanup_remote_tmp and not C.DEFAULT_KEEP_REMOTE_FILES and "-tmp-" in tmp_path

 434     def _remove_tmp_path(self, tmp_path):
 435         '''Remove a temporary path we created. '''
 436 
 437         if tmp_path is None and self._connection._shell.tmpdir:
 438             tmp_path = self._connection._shell.tmpdir
 439 
 440         if self._should_remove_tmp_path(tmp_path):
 441             cmd = self._connection._shell.remove(tmp_path, recurse=True)
 442             # If we have gotten here we have a working ssh configuration.
 443             # If ssh breaks we could leave tmp directories out on the remote system.
 444             tmp_rm_res = self._low_level_execute_command(cmd, sudoable=False)
 445 
 446             if tmp_rm_res.get('rc', 0) != 0:
 447                 display.warning('Error deleting remote temporary files (rc: %s, stderr: %s})'
 448                                 % (tmp_rm_res.get('rc'), tmp_rm_res.get('stderr', 'No error string available.')))
 449             else:
 450                 self._connection._shell.tmpdir = None

 452     def _transfer_file(self, local_path, remote_path):
 453         """
 454         Copy a file from the controller to a remote path
 455 
 456         :arg local_path: Path on controller to transfer
 457         :arg remote_path: Path on the remote system to transfer into
 458 
 459         .. warning::
 460             * When you use this function you likely want to use use fixup_perms2() on the
 461               remote_path to make sure that the remote file is readable when the user becomes
 462               a non-privileged user.
 463             * If you use fixup_perms2() on the file and copy or move the file into place, you will
 464               need to then remove filesystem acls on the file once it has been copied into place by
 465               the module.  See how the copy module implements this for help.
 466         """
 467         self._connection.put_file(local_path, remote_path)
 468         return remote_path

 470     def _transfer_data(self, remote_path, data):
 471         '''
 472         Copies the module data out to the temporary module path.
 473         '''
 474 
 475         if isinstance(data, dict):
 476             data = jsonify(data)
 477 
 478         afd, afile = tempfile.mkstemp(dir=C.DEFAULT_LOCAL_TMP)
 479         afo = os.fdopen(afd, 'wb')
 480         try:
 481             data = to_bytes(data, errors='surrogate_or_strict')
 482             afo.write(data)
 483         except Exception as e:
 484             raise AnsibleError("failure writing module data to temporary file for transfer: %s" % to_native(e))
 485 
 486         afo.flush()
 487         afo.close()
 488 
 489         try:
 490             self._transfer_file(afile, remote_path)
 491         finally:
 492             os.unlink(afile)
 493 
 494         return remote_path

 496     def _fixup_perms2(self, remote_paths, remote_user=None, execute=True):
 497         """
 498         We need the files we upload to be readable (and sometimes executable)
 499         by the user being sudo'd to but we want to limit other people's access
 500         (because the files could contain passwords or other private
 501         information.  We achieve this in one of these ways:
 502 
 503         * If no sudo is performed or the remote_user is sudo'ing to
 504           themselves, we don't have to change permissions.
 505         * If the remote_user sudo's to a privileged user (for instance, root),
 506           we don't have to change permissions
 507         * If the remote_user sudo's to an unprivileged user then we attempt to
 508           grant the unprivileged user access via file system acls.
 509         * If granting file system acls fails we try to change the owner of the
 510           file with chown which only works in case the remote_user is
 511           privileged or the remote systems allows chown calls by unprivileged
 512           users (e.g. HP-UX)
 513         * If the chown fails, we check if ansible_common_remote_group is set.
 514           If it is, we attempt to chgrp the file to its value. This is useful
 515           if the remote_user has a group in common with the become_user. As the
 516           remote_user, we can chgrp the file to that group and allow the
 517           become_user to read it.
 518         * If (the chown fails AND ansible_common_remote_group is not set) OR
 519           (ansible_common_remote_group is set AND the chgrp (or following chmod)
 520           returned non-zero), we can set the file to be world readable so that
 521           the second unprivileged user can read the file.
 522           Since this could allow other users to get access to private
 523           information we only do this if ansible is configured with
 524           "allow_world_readable_tmpfiles" in the ansible.cfg. Also note that
 525           when ansible_common_remote_group is set this final fallback is very
 526           unlikely to ever be triggered, so long as chgrp was successful. But
 527           just because the chgrp was successful, does not mean Ansible can
 528           necessarily access the files (if, for example, the variable was set
 529           to a group that remote_user is in, and can chgrp to, but does not have
 530           in common with become_user).
 531         """
 532         if remote_user is None:
 533             remote_user = self._get_remote_user()
 534 
 535         if getattr(self._connection._shell, "_IS_WINDOWS", False):
 536             # This won't work on Powershell as-is, so we'll just completely skip until
 537             # we have a need for it, at which point we'll have to do something different.
 538             return remote_paths
 539 
 540         if self._is_become_unprivileged():
 541             # Unprivileged user that's different than the ssh user.  Let's get
 542             # to work!
 543 
 544             become_user = self.get_become_option('become_user')
 545 
 546             # Try to use file system acls to make the files readable for sudo'd
 547             # user
 548             if execute:
 549                 chmod_mode = 'rx'
 550                 setfacl_mode = 'r-x'
 551             else:
 552                 chmod_mode = 'rX'
 553                 # NOTE: this form fails silently on freebsd.  We currently
 554                 # never call _fixup_perms2() with execute=False but if we
 555                 # start to we'll have to fix this.
 556                 setfacl_mode = 'r-X'
 557 
 558             res = self._remote_set_user_facl(remote_paths, become_user, setfacl_mode)
 559             if res['rc'] != 0:
 560                 # File system acls failed; let's try to use chown next
 561                 # Set executable bit first as on some systems an
 562                 # unprivileged user can use chown
 563                 if execute:
 564                     res = self._remote_chmod(remote_paths, 'u+x')
 565                     if res['rc'] != 0:
 566                         raise AnsibleError('Failed to set file mode on remote temporary files (rc: {0}, err: {1})'.format(res['rc'], to_native(res['stderr'])))
 567 
 568                 res = self._remote_chown(remote_paths, become_user)
 569                 if res['rc'] != 0:
 570                     # First check if we are an admin/root user. If we are
 571                     # and failed here, something weird has happened.
 572                     if remote_user in self._get_admin_users():
 573                         # chown failed even if remote_user is administrator/root
 574                         raise AnsibleError('Failed to change ownership of the temporary files Ansible needs to create despite connecting as a privileged user. '
 575                                            'Unprivileged become user would be unable to read the file.')
 576 
 577                     # Otherwise, we're a normal user. We failed to chown the
 578                     # paths to the unprivileged user, but if we have a common
 579                     # group with them, we should be able to chown it to that.
 580                     #
 581                     # Note that we have no way of knowing if this will actually
 582                     # work... just because chgrp exits successfully does not
 583                     # mean that Ansible will work. We could check if the become
 584                     # user is in the group, but this would create an extra
 585                     # round trip.
 586                     #
 587                     # Also note that due to the above, this can prevent the
 588                     # ALLOW_WORLD_READABLE_TMPFILES logic below from ever
 589                     # getting called. We leave this up to the user to rectify
 590                     # if they have both of these features enabled.
 591                     group = self.get_shell_option('common_remote_group')
 592                     if group is not None:
 593                         res = self._remote_chgrp(remote_paths, group)
 594                         if res['rc'] == 0:
 595                             # If ALLOW_WORLD_READABLE_TMPFILES is set, we should warn the user
 596                             # that something might go weirdly here.
 597                             if C.ALLOW_WORLD_READABLE_TMPFILES:
 598                                 display.warning('Both common_remote_group and allow_world_readable_tmpfiles are set. chgrp was successful, but there is no '
 599                                                 'guarantee that Ansible will be able to read the files after this operation, particularly if '
 600                                                 'common_remote_group was set to a group of which the unprivileged become user is not a member. In this '
 601                                                 'situation, allow_world_readable_tmpfiles is a no-op. See the "Risks of becoming an unprivileged user" section '
 602                                                 'of the "Understanding privilege escalation: become" user guide documentation for more information')
 603                             if execute:
 604                                 group_mode = 'g+rwx'
 605                             else:
 606                                 group_mode = 'g+rw'
 607                             res = self._remote_chmod(remote_paths, group_mode)
 608 
 609                 if res['rc'] != 0:
 610                     if self.get_shell_option('world_readable_temp', C.ALLOW_WORLD_READABLE_TMPFILES):
 611                         # chown and fs acls failed -- do things this insecure
 612                         # way only if the user opted in in the config file
 613                         display.warning('Using world-readable permissions for temporary files Ansible needs to create when becoming an unprivileged user. '
 614                                         'This may be insecure. For information on securing this, see '
 615                                         'https://docs.ansible.com/ansible/user_guide/become.html#risks-of-becoming-an-unprivileged-user')
 616                         res = self._remote_chmod(remote_paths, 'a+%s' % chmod_mode)
 617                         if res['rc'] != 0:
 618                             raise AnsibleError('Failed to set file mode on remote files (rc: {0}, err: {1})'.format(res['rc'], to_native(res['stderr'])))
 619                     else:
 620                         raise AnsibleError('Failed to set permissions on the temporary files Ansible needs to create when becoming an unprivileged user '
 621                                            '(rc: %s, err: %s}). For information on working around this, see '
 622                                            'https://docs.ansible.com/ansible/become.html#becoming-an-unprivileged-user'
 623                                            % (res['rc'], to_native(res['stderr'])))
 624         elif execute:
 625             # Can't depend on the file being transferred with execute permissions.
 626             # Only need user perms because no become was used here
 627             res = self._remote_chmod(remote_paths, 'u+x')
 628             if res['rc'] != 0:
 629                 raise AnsibleError('Failed to set execute bit on remote files (rc: {0}, err: {1})'.format(res['rc'], to_native(res['stderr'])))
 630 
 631         return remote_paths

 665     def _execute_remote_stat(self, path, all_vars, follow, tmp=None, checksum=True):
 666         '''
 667         Get information from remote file.
 668         '''
 669         if tmp is not None:
 670             display.warning('_execute_remote_stat no longer honors the tmp parameter. Action'
 671                             ' plugins should set self._connection._shell.tmpdir to share'
 672                             ' the tmpdir')
 673         del tmp  # No longer used
 674 
 675         module_args = dict(
 676             path=path,
 677             follow=follow,
 678             get_checksum=checksum,
 679             checksum_algorithm='sha1',
 680         )
 681         mystat = self._execute_module(module_name='stat', module_args=module_args, task_vars=all_vars,
 682                                       wrap_async=False)
 683 
 684         if mystat.get('failed'):
 685             msg = mystat.get('module_stderr')
 686             if not msg:
 687                 msg = mystat.get('module_stdout')
 688             if not msg:
 689                 msg = mystat.get('msg')
 690             raise AnsibleError('Failed to get information on remote file (%s): %s' % (path, msg))
 691 
 692         if not mystat['stat']['exists']:
 693             # empty might be matched, 1 should never match, also backwards compatible
 694             mystat['stat']['checksum'] = '1'
 695 
 696         # happens sometimes when it is a dir and not on bsd
 697         if 'checksum' not in mystat['stat']:
 698             mystat['stat']['checksum'] = ''
 699         elif not isinstance(mystat['stat']['checksum'], string_types):
 700             raise AnsibleError("Invalid checksum returned by stat: expected a string type but got %s" % type(mystat['stat']['checksum']))
 701 
 702         return mystat['stat']

 704     def _remote_checksum(self, path, all_vars, follow=False):
 705         '''
 706         Produces a remote checksum given a path,
 707         Returns a number 0-4 for specific errors instead of checksum, also ensures it is different
 708         0 = unknown error
 709         1 = file does not exist, this might not be an error
 710         2 = permissions issue
 711         3 = its a directory, not a file
 712         4 = stat module failed, likely due to not finding python
 713         5 = appropriate json module not found
 714         '''
 715         x = "0"  # unknown error has occurred
 716         try:
 717             remote_stat = self._execute_remote_stat(path, all_vars, follow=follow)
 718             if remote_stat['exists'] and remote_stat['isdir']:
 719                 x = "3"  # its a directory not a file
 720             else:
 721                 x = remote_stat['checksum']  # if 1, file is missing
 722         except AnsibleError as e:
 723             errormsg = to_text(e)
 724             if errormsg.endswith(u'Permission denied'):
 725                 x = "2"  # cannot read file
 726             elif errormsg.endswith(u'MODULE FAILURE'):
 727                 x = "4"  # python not found or module uncaught exception
 728             elif 'json' in errormsg:
 729                 x = "5"  # json module needed
 730         finally:
 731             return x  # pylint: disable=lost-exception

 733     def _remote_expand_user(self, path, sudoable=True, pathsep=None):
 734         ''' takes a remote path and performs tilde/$HOME expansion on the remote host '''
 735 
 736         # We only expand ~/path and ~username/path
 737         if not path.startswith('~'):
 738             return path
 739 
 740         # Per Jborean, we don't have to worry about Windows as we don't have a notion of user's home
 741         # dir there.
 742         split_path = path.split(os.path.sep, 1)
 743         expand_path = split_path[0]
 744 
 745         if expand_path == '~':
 746             # Network connection plugins (network_cli, netconf, etc.) execute on the controller, rather than the remote host.
 747             # As such, we want to avoid using remote_user for paths  as remote_user may not line up with the local user
 748             # This is a hack and should be solved by more intelligent handling of remote_tmp in 2.7
 749             become_user = self.get_become_option('become_user')
 750             if getattr(self._connection, '_remote_is_local', False):
 751                 pass
 752             elif sudoable and self._connection.become and become_user:
 753                 expand_path = '~%s' % become_user
 754             else:
 755                 # use remote user instead, if none set default to current user
 756                 expand_path = '~%s' % (self._get_remote_user() or '')
 757 
 758         # use shell to construct appropriate command and execute
 759         cmd = self._connection._shell.expand_user(expand_path)
 760         data = self._low_level_execute_command(cmd, sudoable=False)
 761 
 762         try:
 763             initial_fragment = data['stdout'].strip().splitlines()[-1]
 764         except IndexError:
 765             initial_fragment = None
 766 
 767         if not initial_fragment:
 768             # Something went wrong trying to expand the path remotely. Try using pwd, if not, return
 769             # the original string
 770             cmd = self._connection._shell.pwd()
 771             pwd = self._low_level_execute_command(cmd, sudoable=False).get('stdout', '').strip()
 772             if pwd:
 773                 expanded = pwd
 774             else:
 775                 expanded = path
 776 
 777         elif len(split_path) > 1:
 778             expanded = self._connection._shell.join_path(initial_fragment, *split_path[1:])
 779         else:
 780             expanded = initial_fragment
 781 
 782         if '..' in os.path.dirname(expanded).split('/'):
 783             raise AnsibleError("'%s' returned an invalid relative home directory path containing '..'" % self._play_context.remote_addr)
 784 
 785         return expanded

 853     def _execute_module(self, module_name=None, module_args=None, tmp=None, task_vars=None, persist_files=False, delete_remote_tmp=None, wrap_async=False):
 854         '''
 855         Transfer and run a module along with its arguments.
 856         '''
 857         if tmp is not None:
 858             display.warning('_execute_module no longer honors the tmp parameter. Action plugins'
 859                             ' should set self._connection._shell.tmpdir to share the tmpdir')
 860         del tmp  # No longer used
 861         if delete_remote_tmp is not None:
 862             display.warning('_execute_module no longer honors the delete_remote_tmp parameter.'
 863                             ' Action plugins should check self._connection._shell.tmpdir to'
 864                             ' see if a tmpdir existed before they were called to determine'
 865                             ' if they are responsible for removing it.')
 866         del delete_remote_tmp  # No longer used
 867 
 868         tmpdir = self._connection._shell.tmpdir
 869 
 870         # We set the module_style to new here so the remote_tmp is created
 871         # before the module args are built if remote_tmp is needed (async).
 872         # If the module_style turns out to not be new and we didn't create the
 873         # remote tmp here, it will still be created. This must be done before
 874         # calling self._update_module_args() so the module wrapper has the
 875         # correct remote_tmp value set
 876         if not self._is_pipelining_enabled("new", wrap_async) and tmpdir is None:
 877             self._make_tmp_path()
 878             tmpdir = self._connection._shell.tmpdir
 879 
 880         if task_vars is None:
 881             task_vars = dict()
 882 
 883         # if a module name was not specified for this execution, use the action from the task
 884         if module_name is None:
 885             module_name = self._task.action
 886         if module_args is None:
 887             module_args = self._task.args
 888 
 889         self._update_module_args(module_name, module_args, task_vars)
 890 
 891         # FIXME: convert async_wrapper.py to not rely on environment variables
 892         # make sure we get the right async_dir variable, backwards compatibility
 893         # means we need to lookup the env value ANSIBLE_ASYNC_DIR first
 894         remove_async_dir = None
 895         if wrap_async or self._task.async_val:
 896             env_async_dir = [e for e in self._task.environment if
 897                              "ANSIBLE_ASYNC_DIR" in e]
 898             if len(env_async_dir) > 0:
 899                 msg = "Setting the async dir from the environment keyword " \
 900                       "ANSIBLE_ASYNC_DIR is deprecated. Set the async_dir " \
 901                       "shell option instead"
 902                 self._display.deprecated(msg, "2.12", collection_name='ansible.builtin')
 903             else:
 904                 # ANSIBLE_ASYNC_DIR is not set on the task, we get the value
 905                 # from the shell option and temporarily add to the environment
 906                 # list for async_wrapper to pick up
 907                 async_dir = self.get_shell_option('async_dir', default="~/.ansible_async")
 908                 remove_async_dir = len(self._task.environment)
 909                 self._task.environment.append({"ANSIBLE_ASYNC_DIR": async_dir})
 910 
 911         # FUTURE: refactor this along with module build process to better encapsulate "smart wrapper" functionality
 912         (module_style, shebang, module_data, module_path) = self._configure_module(module_name=module_name, module_args=module_args, task_vars=task_vars)
 913         display.vvv("Using module file %s" % module_path)
 914         if not shebang and module_style != 'binary':
 915             raise AnsibleError("module (%s) is missing interpreter line" % module_name)
 916 
 917         self._used_interpreter = shebang
 918         remote_module_path = None
 919 
 920         if not self._is_pipelining_enabled(module_style, wrap_async):
 921             # we might need remote tmp dir
 922             if tmpdir is None:
 923                 self._make_tmp_path()
 924                 tmpdir = self._connection._shell.tmpdir
 925 
 926             remote_module_filename = self._connection._shell.get_remote_filename(module_path)
 927             remote_module_path = self._connection._shell.join_path(tmpdir, 'AnsiballZ_%s' % remote_module_filename)
 928 
 929         args_file_path = None
 930         if module_style in ('old', 'non_native_want_json', 'binary'):
 931             # we'll also need a tmp file to hold our module arguments
 932             args_file_path = self._connection._shell.join_path(tmpdir, 'args')
 933 
 934         if remote_module_path or module_style != 'new':
 935             display.debug("transferring module to remote %s" % remote_module_path)
 936             if module_style == 'binary':
 937                 self._transfer_file(module_path, remote_module_path)
 938             else:
 939                 self._transfer_data(remote_module_path, module_data)
 940             if module_style == 'old':
 941                 # we need to dump the module args to a k=v string in a file on
 942                 # the remote system, which can be read and parsed by the module
 943                 args_data = ""
 944                 for k, v in iteritems(module_args):
 945                     args_data += '%s=%s ' % (k, shlex_quote(text_type(v)))
 946                 self._transfer_data(args_file_path, args_data)
 947             elif module_style in ('non_native_want_json', 'binary'):
 948                 self._transfer_data(args_file_path, json.dumps(module_args))
 949             display.debug("done transferring module to remote")
 950 
 951         environment_string = self._compute_environment_string()
 952 
 953         # remove the ANSIBLE_ASYNC_DIR env entry if we added a temporary one for
 954         # the async_wrapper task - this is so the async_status plugin doesn't
 955         # fire a deprecation warning when it runs after this task
 956         if remove_async_dir is not None:
 957             del self._task.environment[remove_async_dir]
 958 
 959         remote_files = []
 960         if tmpdir and remote_module_path:
 961             remote_files = [tmpdir, remote_module_path]
 962 
 963         if args_file_path:
 964             remote_files.append(args_file_path)
 965 
 966         sudoable = True
 967         in_data = None
 968         cmd = ""
 969 
 970         if wrap_async and not self._connection.always_pipeline_modules:
 971             # configure, upload, and chmod the async_wrapper module
 972             (async_module_style, shebang, async_module_data, async_module_path) = self._configure_module(module_name='async_wrapper', module_args=dict(),
 973                                                                                                          task_vars=task_vars)
 974             async_module_remote_filename = self._connection._shell.get_remote_filename(async_module_path)
 975             remote_async_module_path = self._connection._shell.join_path(tmpdir, async_module_remote_filename)
 976             self._transfer_data(remote_async_module_path, async_module_data)
 977             remote_files.append(remote_async_module_path)
 978 
 979             async_limit = self._task.async_val
 980             async_jid = str(random.randint(0, 999999999999))
 981 
 982             # call the interpreter for async_wrapper directly
 983             # this permits use of a script for an interpreter on non-Linux platforms
 984             # TODO: re-implement async_wrapper as a regular module to avoid this special case
 985             interpreter = shebang.replace('#!', '').strip()
 986             async_cmd = [interpreter, remote_async_module_path, async_jid, async_limit, remote_module_path]
 987 
 988             if environment_string:
 989                 async_cmd.insert(0, environment_string)
 990 
 991             if args_file_path:
 992                 async_cmd.append(args_file_path)
 993             else:
 994                 # maintain a fixed number of positional parameters for async_wrapper
 995                 async_cmd.append('_')
 996 
 997             if not self._should_remove_tmp_path(tmpdir):
 998                 async_cmd.append("-preserve_tmp")
 999 
1000             cmd = " ".join(to_text(x) for x in async_cmd)
1001 
1002         else:
1003 
1004             if self._is_pipelining_enabled(module_style):
1005                 in_data = module_data
1006                 display.vvv("Pipelining is enabled.")
1007             else:
1008                 cmd = remote_module_path
1009 
1010             cmd = self._connection._shell.build_module_command(environment_string, shebang, cmd, arg_path=args_file_path).strip()
1011 
1012         # Fix permissions of the tmpdir path and tmpdir files. This should be called after all
1013         # files have been transferred.
1014         if remote_files:
1015             # remove none/empty
1016             remote_files = [x for x in remote_files if x]
1017             self._fixup_perms2(remote_files, self._get_remote_user())
1018 
1019         # actually execute
1020         res = self._low_level_execute_command(cmd, sudoable=sudoable, in_data=in_data)
1021 
1022         # parse the main result
1023         data = self._parse_returned_data(res)
1024 
1025         # NOTE: INTERNAL KEYS ONLY ACCESSIBLE HERE
1026         # get internal info before cleaning
1027         if data.pop("_ansible_suppress_tmpdir_delete", False):
1028             self._cleanup_remote_tmp = False
1029 
1030         # NOTE: yum returns results .. but that made it 'compatible' with squashing, so we allow mappings, for now
1031         if 'results' in data and (not isinstance(data['results'], Sequence) or isinstance(data['results'], string_types)):
1032             data['ansible_module_results'] = data['results']
1033             del data['results']
1034             display.warning("Found internal 'results' key in module return, renamed to 'ansible_module_results'.")
1035 
1036         # remove internal keys
1037         remove_internal_keys(data)
1038 
1039         if wrap_async:
1040             # async_wrapper will clean up its tmpdir on its own so we want the controller side to
1041             # forget about it now
1042             self._connection._shell.tmpdir = None
1043 
1044             # FIXME: for backwards compat, figure out if still makes sense
1045             data['changed'] = True
1046 
1047         # pre-split stdout/stderr into lines if needed
1048         if 'stdout' in data and 'stdout_lines' not in data:
1049             # if the value is 'False', a default won't catch it.
1050             txt = data.get('stdout', None) or u''
1051             data['stdout_lines'] = txt.splitlines()
1052         if 'stderr' in data and 'stderr_lines' not in data:
1053             # if the value is 'False', a default won't catch it.
1054             txt = data.get('stderr', None) or u''
1055             data['stderr_lines'] = txt.splitlines()
1056

1058         if self._discovered_interpreter_key:
1059             if data.get('ansible_facts') is None:
1060                 data['ansible_facts'] = {}
1061 
1062             data['ansible_facts'][self._discovered_interpreter_key] = self._discovered_interpreter
1063 
1064         if self._discovery_warnings:
1065             if data.get('warnings') is None:
1066                 data['warnings'] = []
1067             data['warnings'].extend(self._discovery_warnings)
1068 
1069         if self._discovery_deprecation_warnings:
1070             if data.get('deprecations') is None:
1071                 data['deprecations'] = []
1072             data['deprecations'].extend(self._discovery_deprecation_warnings)
1073 
1074         # mark the entire module results untrusted as a template right here, since the current action could
1075         # possibly template one of these values.
1076         data = wrap_var(data)
1077 
1078         display.debug("done with _execute_module (%s, %s)" % (module_name, module_args))
1079         return data

1081     def _parse_returned_data(self, res):
1082         try:
1083             filtered_output, warnings = _filter_non_json_lines(res.get('stdout', u''))
1084             for w in warnings:
1085                 display.warning(w)
1086 
1087             data = json.loads(filtered_output)
1088             data['_ansible_parsed'] = True
1089         except ValueError:
1090             # not valid json, lets try to capture error
1091             data = dict(failed=True, _ansible_parsed=False)
1092             data['module_stdout'] = res.get('stdout', u'')
1093             if 'stderr' in res:
1094                 data['module_stderr'] = res['stderr']
1095                 if res['stderr'].startswith(u'Traceback'):
1096                     data['exception'] = res['stderr']
1097 
1098             # in some cases a traceback will arrive on stdout instead of stderr, such as when using ssh with -tt
1099             if 'exception' not in data and data['module_stdout'].startswith(u'Traceback'):
1100                 data['exception'] = data['module_stdout']
1101 
1102             # The default
1103             data['msg'] = "MODULE FAILURE"
1104 
1105             # try to figure out if we are missing interpreter
1106             if self._used_interpreter is not None:
1107                 match = re.compile('%s: (?:No such file or directory|not found)' % self._used_interpreter.lstrip('!#'))
1108                 if match.search(data['module_stderr']) or match.search(data['module_stdout']):
1109                     data['msg'] = "The module failed to execute correctly, you probably need to set the interpreter."
1110 
1111             # always append hint
1112             data['msg'] += '\nSee stdout/stderr for the exact error'
1113 
1114             if 'rc' in res:
1115                 data['rc'] = res['rc']
1116         return data

1119     def _low_level_execute_command(self, cmd, sudoable=True, in_data=None, executable=None, encoding_errors='surrogate_then_replace', chdir=None):
1120         '''
1121         This is the function which executes the low level shell command, which
1122         may be commands to create/remove directories for temporary files, or to
1123         run the module code or python directly when pipelining.
1124 
1125         :kwarg encoding_errors: If the value returned by the command isn't
1126             utf-8 then we have to figure out how to transform it to unicode.
1127             If the value is just going to be displayed to the user (or
1128             discarded) then the default of 'replace' is fine.  If the data is
1129             used as a key or is going to be written back out to a file
1130             verbatim, then this won't work.  May have to use some sort of
1131             replacement strategy (python3 could use surrogateescape)
1132         :kwarg chdir: cd into this directory before executing the command.
1133         '''
1134 
1135         display.debug("_low_level_execute_command(): starting")
1136         # if not cmd:
1137         #     # this can happen with powershell modules when there is no analog to a Windows command (like chmod)
1138         #     display.debug("_low_level_execute_command(): no command, exiting")
1139         #     return dict(stdout='', stderr='', rc=254)
1140 
1141         if chdir:
1142             display.debug("_low_level_execute_command(): changing cwd to %s for this command" % chdir)
1143             cmd = self._connection._shell.append_command('cd %s' % chdir, cmd)
1144 
1145         # https://github.com/ansible/ansible/issues/68054
1146         if executable:
1147             self._connection._shell.executable = executable
1148 
1149         ruser = self._get_remote_user()
1150         buser = self.get_become_option('become_user')
1151         if (sudoable and self._connection.become and  # if sudoable and have become
1152                 resource_from_fqcr(self._connection.transport) != 'network_cli' and  # if not using network_cli
1153                 (C.BECOME_ALLOW_SAME_USER or (buser != ruser or not any((ruser, buser))))):  # if we allow same user PE or users are different and either is set
1154             display.debug("_low_level_execute_command(): using become for this command")
1155             cmd = self._connection.become.build_become_command(cmd, self._connection._shell)
1156 
1157         if self._connection.allow_executable:
1158             if executable is None:
1159                 executable = self._play_context.executable
1160                 # mitigation for SSH race which can drop stdout (https://github.com/ansible/ansible/issues/13876)
1161                 # only applied for the default executable to avoid interfering with the raw action
1162                 cmd = self._connection._shell.append_command(cmd, 'sleep 0')
1163             if executable:
1164                 cmd = executable + ' -c ' + shlex_quote(cmd)
1165 
1166         display.debug("_low_level_execute_command(): executing: %s" % (cmd,))
1167 
1168         # Change directory to basedir of task for command execution when connection is local
1169         if self._connection.transport == 'local':
1170             self._connection.cwd = to_bytes(self._loader.get_basedir(), errors='surrogate_or_strict')
1171 
1172         rc, stdout, stderr = self._connection.exec_command(cmd, in_data=in_data, sudoable=sudoable)
1173 
1174         # stdout and stderr may be either a file-like or a bytes object.
1175         # Convert either one to a text type
1176         if isinstance(stdout, binary_type):
1177             out = to_text(stdout, errors=encoding_errors)
1178         elif not isinstance(stdout, text_type):
1179             out = to_text(b''.join(stdout.readlines()), errors=encoding_errors)
1180         else:
1181             out = stdout
1182 
1183         if isinstance(stderr, binary_type):
1184             err = to_text(stderr, errors=encoding_errors)
1185         elif not isinstance(stderr, text_type):
1186             err = to_text(b''.join(stderr.readlines()), errors=encoding_errors)
1187         else:
1188             err = stderr
1189 
1190         if rc is None:
1191             rc = 0
1192 
1193         # be sure to remove the BECOME-SUCCESS message now
1194         out = self._strip_success_message(out)
1195 
1196         display.debug(u"_low_level_execute_command() done: rc=%d, stdout=%s, stderr=%s" % (rc, out, err))
1197         return dict(rc=rc, stdout=out, stdout_lines=out.splitlines(), stderr=err, stderr_lines=err.splitlines())

1199     def _get_diff_data(self, destination, source, task_vars, source_file=True):
1200 
1201         # Note: Since we do not diff the source and destination before we transform from bytes into
1202         # text the diff between source and destination may not be accurate.  To fix this, we'd need
1203         # to move the diffing from the callback plugins into here.
1204         #
1205         # Example of data which would cause trouble is src_content == b'\xff' and dest_content ==
1206         # b'\xfe'.  Neither of those are valid utf-8 so both get turned into the replacement
1207         # character: diff['before'] = u'' ; diff['after'] = u''  When the callback plugin later
1208         # diffs before and after it shows an empty diff.
1209 
1210         diff = {}
1211         display.debug("Going to peek to see if file has changed permissions")
1212         peek_result = self._execute_module(module_name='file', module_args=dict(path=destination, _diff_peek=True), task_vars=task_vars, persist_files=True)
1213 
1214         if peek_result.get('failed', False):
1215             display.warning(u"Failed to get diff between '%s' and '%s': %s" % (os.path.basename(source), destination, to_text(peek_result.get(u'msg', u''))))
1216             return diff
1217 
1218         if peek_result.get('rc', 0) == 0:
1219 
1220             if peek_result.get('state') in (None, 'absent'):
1221                 diff['before'] = u''
1222             elif peek_result.get('appears_binary'):
1223                 diff['dst_binary'] = 1
1224             elif peek_result.get('size') and C.MAX_FILE_SIZE_FOR_DIFF > 0 and peek_result['size'] > C.MAX_FILE_SIZE_FOR_DIFF:
1225                 diff['dst_larger'] = C.MAX_FILE_SIZE_FOR_DIFF
1226             else:
1227                 display.debug(u"Slurping the file %s" % source)
1228                 dest_result = self._execute_module(module_name='slurp', module_args=dict(path=destination), task_vars=task_vars, persist_files=True)
1229                 if 'content' in dest_result:
1230                     dest_contents = dest_result['content']
1231                     if dest_result['encoding'] == u'base64':
1232                         dest_contents = base64.b64decode(dest_contents)
1233                     else:
1234                         raise AnsibleError("unknown encoding in content option, failed: %s" % to_native(dest_result))
1235                     diff['before_header'] = destination
1236                     diff['before'] = to_text(dest_contents)
1237 
1238             if source_file:
1239                 st = os.stat(source)
1240                 if C.MAX_FILE_SIZE_FOR_DIFF > 0 and st[stat.ST_SIZE] > C.MAX_FILE_SIZE_FOR_DIFF:
1241                     diff['src_larger'] = C.MAX_FILE_SIZE_FOR_DIFF
1242                 else:
1243                     display.debug("Reading local copy of the file %s" % source)
1244                     try:
1245                         with open(source, 'rb') as src:
1246                             src_contents = src.read()
1247                     except Exception as e:
1248                         raise AnsibleError("Unexpected error while reading source (%s) for diff: %s " % (source, to_native(e)))
1249 
1250                     if b"\x00" in src_contents:
1251                         diff['src_binary'] = 1
1252                     else:
1253                         diff['after_header'] = source
1254                         diff['after'] = to_text(src_contents)
1255             else:
1256                 display.debug(u"source of file passed in")
1257                 diff['after_header'] = u'dynamically generated'
1258                 diff['after'] = source
1259 
1260         if self._play_context.no_log:
1261             if 'before' in diff:
1262                 diff["before"] = u""
1263             if 'after' in diff:
1264                 diff["after"] = u" [[ Diff output has been hidden because 'no_log: true' was specified for this result ]]\n"
1265 
1266         return diff
