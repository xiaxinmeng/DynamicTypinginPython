  61 def get_relative_path(requests_pathname, path):
  62     if requests_pathname == "/" and path == "":
  63         return "/"
  64     elif requests_pathname != "/" and path == "":
  65         return requests_pathname
  66     elif not path.startswith("/"):
  67         raise exceptions.UnsupportedRelativePath(
  68             "Paths that aren't prefixed with a leading / are not supported.\n"
  69             + "You supplied: {}".format(path)
  70         )
  71     return "/".join([requests_pathname.rstrip("/"), path.lstrip("/")])

  74 def strip_relative_path(requests_pathname, path):
  75     if path is None:
  76         return None
  77     elif (
  78         requests_pathname != "/" and not path.startswith(requests_pathname.rstrip("/"))
  79     ) or (requests_pathname == "/" and not path.startswith("/")):
  80         raise exceptions.UnsupportedRelativePath(
  81             "Paths that aren't prefixed with a leading "
  82             + "requests_pathname_prefix are not supported.\n"
  83             + "You supplied: {} and requests_pathname_prefix was {}".format(
  84                 path, requests_pathname
  85             )
  86         )
  87     if requests_pathname != "/" and path.startswith(requests_pathname.rstrip("/")):
  88         path = path.replace(
  89             # handle the case where the path might be `/my-dash-app`
  90             # but the requests_pathname_prefix is `/my-dash-app/`
  91             requests_pathname.rstrip("/"),
  92             "",
  93             1,
  94         )
  95     return path.strip("/")

 103 class AttributeDict(dict):
 104     """Dictionary subclass enabling attribute lookup/assignment of keys/values.
 105 
 106     For example::
 107         >>> m = AttributeDict({'foo': 'bar'})
 108         >>> m.foo
 109         'bar'
 110         >>> m.foo = 'not bar'
 111         >>> m['foo']
 112         'not bar'
 113     ``AttributeDict`` objects also provide ``.first()`` which acts like
 114     ``.get()`` but accepts multiple keys as arguments, and returns the value of
 115     the first hit, e.g.::
 116         >>> m = AttributeDict({'foo': 'bar', 'biz': 'baz'})
 117         >>> m.first('wrong', 'incorrect', 'foo', 'biz')
 118         'bar'
 119     """
 120 
 121     def __setattr__(self, key, value):
 122         self[key] = value
 123 
 124     def __getattr__(self, key):
 125         try:
 126             return self[key]
 127         except KeyError:
 128             pass
 129         # to conform with __getattr__ spec
 130         # but get out of the except block so it doesn't look like a nested err
 131         raise AttributeError(key)
 132 
 133     def set_read_only(self, names, msg="Attribute is read-only"):
 134         object.__setattr__(self, "_read_only", names)
 135         object.__setattr__(self, "_read_only_msg", msg)
 136 
 137     def finalize(self, msg="Object is final: No new keys may be added."):
 138         """Prevent any new keys being set."""
 139         object.__setattr__(self, "_final", msg)
 140 
 141     def __setitem__(self, key, val):
 142         if key in self.__dict__.get("_read_only", []):
 143             raise AttributeError(self._read_only_msg, key)
 144 
 145         final_msg = self.__dict__.get("_final")
 146         if final_msg and key not in self:
 147             raise AttributeError(final_msg, key)
 148 
 149         return super(AttributeDict, self).__setitem__(key, val)
 150 
 151     # pylint: disable=inconsistent-return-statements
 152     def first(self, *names):
 153         for name in names:
 154             value = self.get(name)
 155             if value:
 156                 return value

 159 def create_callback_id(output):
 160     if isinstance(output, (list, tuple)):
 161         return "..{}..".format(
 162             "...".join(
 163                 "{}.{}".format(
 164                     # A single dot within a dict id key or value is OK
 165                     # but in case of multiple dots together escape each dot
 166                     # with `\` so we don't mistake it for multi-outputs
 167                     x.component_id_str().replace(".", "\\."),
 168                     x.component_property,
 169                 )
 170                 for x in output
 171             )
 172         )
 173 
 174     return "{}.{}".format(
 175         output.component_id_str().replace(".", "\\."), output.component_property
 176     )

 182 def split_callback_id(callback_id):
 183     if callback_id.startswith(".."):
 184         return [split_callback_id(oi) for oi in callback_id[2:-2].split("...")]
 185 
 186     id_, prop = callback_id.rsplit(".", 1)
 187     return {"id": id_, "property": prop}

 196 def inputs_to_dict(inputs_list):
 197     inputs = {}
 198     for i in inputs_list:
 199         inputsi = i if isinstance(i, list) else [i]
 200         for ii in inputsi:
 201             id_str = stringify_id(ii["id"])
 202             inputs["{}.{}".format(id_str, ii["property"])] = ii.get("value")
 203     return inputs

 213 def run_command_with_process(cmd):
 214     is_win = sys.platform == "win32"
 215     proc = subprocess.Popen(shlex.split(cmd, posix=is_win), shell=is_win)
 216     proc.wait()
 217     if proc.poll() is None:
 218         logger.warning(" trying to terminate subprocess in safe way")
 219         try:
 220             proc.communicate()
 221         except Exception:  # pylint: disable=broad-except
 222             logger.exception(" first try communicate failed")
 223             proc.kill()
 224             proc.communicate()

 232 def job(msg=""):
 233     def wrapper(func):
 234         @wraps(func)
 235         def _wrapper(*args, **kwargs):
 236             logger.info("  [%s]   - %s", func.__name__, msg)
 237             res = func(*args, **kwargs)
 238             logger.info(":::  [%s] job done  :::", func.__name__)
 239             return res
 240 
 241         return _wrapper
 242 
 243     return wrapper
