   8 def _assert_almost_equal_both(a, b, **kwargs):
   9     """
  10     Check that two objects are approximately equal.
  11 
  12     This check is performed commutatively.
  13 
  14     Parameters
  15     ----------
  16     a : object
  17         The first object to compare.
  18     b : object
  19         The second object to compare.
  20     kwargs : dict
  21         The arguments passed to `tm.assert_almost_equal`.
  22     """
  23     tm.assert_almost_equal(a, b, **kwargs)
  24     tm.assert_almost_equal(b, a, **kwargs)
  25

  27 def _assert_not_almost_equal(a, b, **kwargs):
  28     """
  29     Check that two objects are not approximately equal.
  30 
  31     Parameters
  32     ----------
  33     a : object
  34         The first object to compare.
  35     b : object
  36         The second object to compare.
  37     kwargs : dict
  38         The arguments passed to `tm.assert_almost_equal`.
  39     """
  40     try:
  41         tm.assert_almost_equal(a, b, **kwargs)
  42         msg = f"{a} and {b} were approximately equal when they shouldn't have been"
  43         pytest.fail(msg=msg)
  44     except AssertionError:
  45         pass
