  83 @pytest.mark.parametrize("filename", [None, "test.{ext}"])
  84 def test_compression(parser_and_data, compression_only, buffer, filename):
  85     parser, data, expected = parser_and_data
  86     compress_type = compression_only
  87 
  88     ext = "gz" if compress_type == "gzip" else compress_type
  89     filename = filename if filename is None else filename.format(ext=ext)
  90 
  91     if filename and buffer:
  92         pytest.skip("Cannot deduce compression from buffer of compressed data.")
  93 
  94     with tm.ensure_clean(filename=filename) as path:
  95         tm.write_to_compressed(compress_type, path, data)
  96         compression = "infer" if filename else compress_type
  97 
  98         if buffer:
  99             with open(path, "rb") as f:
 100                 result = parser.read_csv(f, compression=compression)
 101         else:
 102             result = parser.read_csv(path, compression=compression)
 103 
 104         tm.assert_frame_equal(result, expected)
