  23 def _clean_dict(d):
  24     """
  25     Sanitize dictionary for JSON by converting all keys to strings.
  26 
  27     Parameters
  28     ----------
  29     d : dict
  30         The dictionary to convert.
  31 
  32     Returns
  33     -------
  34     cleaned_dict : dict
  35     """
  36     return {str(k): v for k, v in d.items()}

  51 def get_int32_compat_dtype(numpy, orient):
  52     # See GH#32527
  53     dtype = np.int64
  54     if not ((numpy is None or orient == "index") or (numpy is True and orient is None)):
  55         if compat.is_platform_windows():
  56             dtype = np.int32
  57         else:
  58             dtype = np.intp
  59 
  60     return dtype
