  96 def _groupby_and_merge(by, on, left: "DataFrame", right: "DataFrame", merge_pieces):
  97     """
  98     groupby & merge; we are always performing a left-by type operation
  99 
 100     Parameters
 101     ----------
 102     by: field to group
 103     on: duplicates field
 104     left: DataFrame
 105     right: DataFrame
 106     merge_pieces: function for merging
 107     """
 108     pieces = []
 109     if not isinstance(by, (list, tuple)):
 110         by = [by]
 111 
 112     lby = left.groupby(by, sort=False)
 113     rby: Optional[groupby.DataFrameGroupBy] = None
 114 
 115     # if we can groupby the rhs
 116     # then we can get vastly better perf
 117 
 118     try:
 119         rby = right.groupby(by, sort=False)
 120     except KeyError:
 121         pass
 122 
 123     for key, lhs in lby:
 124 
 125         if rby is None:
 126             rhs = right
 127         else:
 128             try:
 129                 rhs = right.take(rby.indices[key])
 130             except KeyError:
 131                 # key doesn't exist in left
 132                 lcols = lhs.columns.tolist()
 133                 cols = lcols + [r for r in right.columns if r not in set(lcols)]
 134                 merged = lhs.reindex(columns=cols)
 135                 merged.index = range(len(merged))
 136                 pieces.append(merged)
 137                 continue
 138 
 139         merged = merge_pieces(lhs, rhs)
 140 
 141         # make sure join keys are in the merged
 142         # TODO, should merge_pieces do this?
 143         for k in by:
 144             if k in merged:
 145                 merged[k] = key
 146 
 147         pieces.append(merged)
 148 
 149     # preserve the original order
 150     # if we have a missing piece this can be reset
 151     from pandas.core.reshape.concat import concat
 152 
 153     result = concat(pieces, ignore_index=True)
 154     result = result.reindex(columns=pieces[0].columns, copy=False)
 155     return result, lby

 158 def merge_ordered(
 159     left,
 160     right,
 161     on=None,
 162     left_on=None,
 163     right_on=None,
 164     left_by=None,
 165     right_by=None,
 166     fill_method=None,
 167     suffixes=("_x", "_y"),
 168     how: str = "outer",
 169 ) -> "DataFrame":
 170     """
 171     Perform merge with optional filling/interpolation.
 172 
 173     Designed for ordered data like time series data. Optionally
 174     perform group-wise merge (see examples).
 175 
 176     Parameters
 177     ----------
 178     left : DataFrame
 179     right : DataFrame
 180     on : label or list
 181         Field names to join on. Must be found in both DataFrames.
 182     left_on : label or list, or array-like
 183         Field names to join on in left DataFrame. Can be a vector or list of
 184         vectors of the length of the DataFrame to use a particular vector as
 185         the join key instead of columns.
 186     right_on : label or list, or array-like
 187         Field names to join on in right DataFrame or vector/list of vectors per
 188         left_on docs.
 189     left_by : column name or list of column names
 190         Group left DataFrame by group columns and merge piece by piece with
 191         right DataFrame.
 192     right_by : column name or list of column names
 193         Group right DataFrame by group columns and merge piece by piece with
 194         left DataFrame.
 195     fill_method : {'ffill', None}, default None
 196         Interpolation method for data.
 197     suffixes : Sequence, default is ("_x", "_y")
 198         A length-2 sequence where each element is optionally a string
 199         indicating the suffix to add to overlapping column names in
 200         `left` and `right` respectively. Pass a value of `None` instead
 201         of a string to indicate that the column name from `left` or
 202         `right` should be left as-is, with no suffix. At least one of the
 203         values must not be None.
 204 
 205         .. versionchanged:: 0.25.0
 206     how : {'left', 'right', 'outer', 'inner'}, default 'outer'
 207         * left: use only keys from left frame (SQL: left outer join)
 208         * right: use only keys from right frame (SQL: right outer join)
 209         * outer: use union of keys from both frames (SQL: full outer join)
 210         * inner: use intersection of keys from both frames (SQL: inner join).
 211 
 212     Returns
 213     -------
 214     DataFrame
 215         The merged DataFrame output type will the be same as
 216         'left', if it is a subclass of DataFrame.
 217 
 218     See Also
 219     --------
 220     merge : Merge with a database-style join.
 221     merge_asof : Merge on nearest keys.
 222 
 223     Examples
 224     --------
 225     >>> df1 = pd.DataFrame(
 226     ...     {
 227     ...         "key": ["a", "c", "e", "a", "c", "e"],
 228     ...         "lvalue": [1, 2, 3, 1, 2, 3],
 229     ...         "group": ["a", "a", "a", "b", "b", "b"]
 230     ...     }
 231     ... )
 232     >>> df1
 233           key  lvalue group
 234     0   a       1     a
 235     1   c       2     a
 236     2   e       3     a
 237     3   a       1     b
 238     4   c       2     b
 239     5   e       3     b
 240 
 241     >>> df2 = pd.DataFrame({"key": ["b", "c", "d"], "rvalue": [1, 2, 3]})
 242     >>> df2
 243           key  rvalue
 244     0   b       1
 245     1   c       2
 246     2   d       3
 247 
 248     >>> merge_ordered(df1, df2, fill_method="ffill", left_by="group")
 249       key  lvalue group  rvalue
 250     0   a       1     a     NaN
 251     1   b       1     a     1.0
 252     2   c       2     a     2.0
 253     3   d       2     a     3.0
 254     4   e       3     a     3.0
 255     5   a       1     b     NaN
 256     6   b       1     b     1.0
 257     7   c       2     b     2.0
 258     8   d       2     b     3.0
 259     9   e       3     b     3.0
 260     """
 261 
 262     def _merger(x, y):
 263         # perform the ordered merge operation
 264         op = _OrderedMerge(
 265             x,
 266             y,
 267             on=on,
 268             left_on=left_on,
 269             right_on=right_on,
 270             suffixes=suffixes,
 271             fill_method=fill_method,
 272             how=how,
 273         )
 274         return op.get_result()
 275 
 276     if left_by is not None and right_by is not None:
 277         raise ValueError("Can only group either left or right frames")
 278     elif left_by is not None:
 279         result, _ = _groupby_and_merge(
 280             left_by, on, left, right, lambda x, y: _merger(x, y)
 281         )
 282     elif right_by is not None:
 283         result, _ = _groupby_and_merge(
 284             right_by, on, right, left, lambda x, y: _merger(y, x)
 285         )
 286     else:
 287         result = _merger(left, right)
 288     return result

 291 def merge_asof(
 292     left,
 293     right,
 294     on=None,
 295     left_on=None,
 296     right_on=None,
 297     left_index: bool = False,
 298     right_index: bool = False,
 299     by=None,
 300     left_by=None,
 301     right_by=None,
 302     suffixes=("_x", "_y"),
 303     tolerance=None,
 304     allow_exact_matches: bool = True,
 305     direction: str = "backward",
 306 ) -> "DataFrame":
 307     """
 308     Perform an asof merge.
 309 
 310     This is similar to a left-join except that we match on nearest
 311     key rather than equal keys. Both DataFrames must be sorted by the key.
 312 
 313     For each row in the left DataFrame:
 314 
 315       - A "backward" search selects the last row in the right DataFrame whose
 316         'on' key is less than or equal to the left's key.
 317 
 318       - A "forward" search selects the first row in the right DataFrame whose
 319         'on' key is greater than or equal to the left's key.
 320 
 321       - A "nearest" search selects the row in the right DataFrame whose 'on'
 322         key is closest in absolute distance to the left's key.
 323 
 324     The default is "backward" and is compatible in versions below 0.20.0.
 325     The direction parameter was added in version 0.20.0 and introduces
 326     "forward" and "nearest".
 327 
 328     Optionally match on equivalent keys with 'by' before searching with 'on'.
 329 
 330     Parameters
 331     ----------
 332     left : DataFrame
 333     right : DataFrame
 334     on : label
 335         Field name to join on. Must be found in both DataFrames.
 336         The data MUST be ordered. Furthermore this must be a numeric column,
 337         such as datetimelike, integer, or float. On or left_on/right_on
 338         must be given.
 339     left_on : label
 340         Field name to join on in left DataFrame.
 341     right_on : label
 342         Field name to join on in right DataFrame.
 343     left_index : bool
 344         Use the index of the left DataFrame as the join key.
 345     right_index : bool
 346         Use the index of the right DataFrame as the join key.
 347     by : column name or list of column names
 348         Match on these columns before performing merge operation.
 349     left_by : column name
 350         Field names to match on in the left DataFrame.
 351     right_by : column name
 352         Field names to match on in the right DataFrame.
 353     suffixes : 2-length sequence (tuple, list, ...)
 354         Suffix to apply to overlapping column names in the left and right
 355         side, respectively.
 356     tolerance : int or Timedelta, optional, default None
 357         Select asof tolerance within this range; must be compatible
 358         with the merge index.
 359     allow_exact_matches : bool, default True
 360 
 361         - If True, allow matching with the same 'on' value
 362           (i.e. less-than-or-equal-to / greater-than-or-equal-to)
 363         - If False, don't match the same 'on' value
 364           (i.e., strictly less-than / strictly greater-than).
 365 
 366     direction : 'backward' (default), 'forward', or 'nearest'
 367         Whether to search for prior, subsequent, or closest matches.
 368 
 369     Returns
 370     -------
 371     merged : DataFrame
 372 
 373     See Also
 374     --------
 375     merge : Merge with a database-style join.
 376     merge_ordered : Merge with optional filling/interpolation.
 377 
 378     Examples
 379     --------
 380     >>> left = pd.DataFrame({"a": [1, 5, 10], "left_val": ["a", "b", "c"]})

 384     1   5        b
 385     2  10        c
 386

 388     >>> right
 389        a  right_val
 390     0  1          1
 391     1  2          2
 392     2  3          3
 393     3  6          6
 394     4  7          7
 395 
 396     >>> pd.merge_asof(left, right, on="a")
 397         a left_val  right_val
 398     0   1        a          1
 399     1   5        b          3
 400     2  10        c          7
 401 
 402     >>> pd.merge_asof(left, right, on="a", allow_exact_matches=False)
 403         a left_val  right_val
 404     0   1        a        NaN
 405     1   5        b        3.0
 406     2  10        c        7.0
 407 
 408     >>> pd.merge_asof(left, right, on="a", direction="forward")
 409         a left_val  right_val
 410     0   1        a        1.0
 411     1   5        b        6.0
 412     2  10        c        NaN
 413

 415         a left_val  right_val
 416     0   1        a          1
 417     1   5        b          6

 425     1         a

 866     def _get_join_info(self):
 867         left_ax = self.left.axes[self.axis]
 868         right_ax = self.right.axes[self.axis]
 869

 871             join_index, left_indexer, right_indexer = left_ax.join(
 872                 right_ax, how=self.how, return_indexers=True, sort=self.sort
 873             )

 875             join_index, left_indexer, right_indexer = _left_join_on_index(
 876                 left_ax, right_ax, self.left_join_keys, sort=self.sort

 880             join_index, right_indexer, left_indexer = _left_join_on_index(
 881                 right_ax, left_ax, self.right_join_keys, sort=self.sort
 882             )

 885 
 886             if self.right_index:
 887                 if len(self.left) > 0:
 888                     join_index = self._create_join_index(
 889                         self.left.index,
 890                         self.right.index,
 891                         left_indexer,
 892                         right_indexer,
 893                         how="right",
 894                     )
 895                 else:
 896                     join_index = self.right.index.take(right_indexer)
 897                     left_indexer = np.array([-1] * len(join_index))
 898             elif self.left_index:
 899                 if len(self.right) > 0:
 900                     join_index = self._create_join_index(
 901                         self.right.index,
 902                         self.left.index,
 903                         right_indexer,
 904                         left_indexer,
 905                         how="left",
 906                     )
 907                 else:
 908                     join_index = self.left.index.take(left_indexer)
 909                     right_indexer = np.array([-1] * len(join_index))
 910             else:
 911                 join_index = Index(np.arange(len(left_indexer)))
 912 
 913         if len(join_index) == 0:
 914             join_index = join_index.astype(object)
 915         return join_index, left_indexer, right_indexer

 917     def _create_join_index(
 918         self,
 919         index: Index,
 920         other_index: Index,
 921         indexer,
 922         other_indexer,
 923         how: str = "left",
 924     ):
 925         """
 926         Create a join index by rearranging one index to match another
 927 
 928         Parameters
 929         ----------
 930         index: Index being rearranged
 931         other_index: Index used to supply values not found in index
 932         indexer: how to rearrange index
 933         how: replacement is only necessary if indexer based on other_index
 934 
 935         Returns
 936         -------
 937         join_index
 938         """
 939         if self.how in (how, "outer") and not isinstance(other_index, MultiIndex):
 940             # if final index requires values in other_index but not target
 941             # index, indexer may hold missing (-1) values, causing Index.take
 942             # to take the final value in target index. So, we set the last
 943             # element to be the desired fill value. We do not use allow_fill
 944             # and fill_value because it throws a ValueError on integer indices
 945             mask = indexer == -1
 946             if np.any(mask):
 947                 fill_value = na_value_for_dtype(index.dtype, compat=False)
 948                 index = index.append(Index([fill_value]))
 949         return index.take(indexer)

 951     def _get_merge_keys(self):
 952         """
 953         Note: has side effects (copy/delete key columns)
 954 
 955         Parameters
 956         ----------
 957         left
 958         right
 959         on
 960 
 961         Returns
 962         -------
 963         left_keys, right_keys
 964         """
 965         left_keys = []
 966         right_keys = []
 967         join_names = []
 968         right_drop = []
 969         left_drop = []
 970 
 971         left, right = self.left, self.right
 972 
 973         is_lkey = lambda x: is_array_like(x) and len(x) == len(left)
 974         is_rkey = lambda x: is_array_like(x) and len(x) == len(right)
 975

 977         # user could, for example, request 'left_index' and 'left_by'. In a

 979         # 'left_on'. (Instead, users have a MultiIndex). That means the
 980         # self.left_on in this function is always empty in a pd.merge(), but
 981         # a pd.merge_asof(left_index=True, left_by=...) will result in a
 982         # self.left_on array with a None in the middle of it. This requires
 983         # a work-around as designated in the code below.
 984         # See _validate_specification() for where this happens.
 985 
 986         # ugh, spaghetti re #733
 987         if _any(self.left_on) and _any(self.right_on):
 988             for lk, rk in zip(self.left_on, self.right_on):
 989                 if is_lkey(lk):
 990                     left_keys.append(lk)
 991                     if is_rkey(rk):
 992                         right_keys.append(rk)
 993                         join_names.append(None)  # what to do?
 994                     else:
 995                         if rk is not None:
 996                             right_keys.append(right._get_label_or_level_values(rk))
 997                             join_names.append(rk)
 998                         else:
 999                             # work-around for merge_asof(right_index=True)
1000                             right_keys.append(right.index)
1001                             join_names.append(right.index.name)
1002                 else:
1003                     if not is_rkey(rk):
1004                         if rk is not None:
1005                             right_keys.append(right._get_label_or_level_values(rk))
1006                         else:
1007                             # work-around for merge_asof(right_index=True)
1008                             right_keys.append(right.index)
1009                         if lk is not None and lk == rk:
1010                             # avoid key upcast in corner case (length-0)
1011                             if len(left) > 0:
1012                                 right_drop.append(rk)
1013                             else:
1014                                 left_drop.append(lk)
1015                     else:
1016                         right_keys.append(rk)
1017                     if lk is not None:
1018                         left_keys.append(left._get_label_or_level_values(lk))
1019                         join_names.append(lk)
1020                     else:
1021                         # work-around for merge_asof(left_index=True)
1022                         left_keys.append(left.index)
1023                         join_names.append(left.index.name)
1024         elif _any(self.left_on):
1025             for k in self.left_on:
1026                 if is_lkey(k):
1027                     left_keys.append(k)
1028                     join_names.append(None)
1029                 else:
1030                     left_keys.append(left._get_label_or_level_values(k))
1031                     join_names.append(k)
1032             if isinstance(self.right.index, MultiIndex):
1033                 right_keys = [
1034                     lev._values.take(lev_codes)
1035                     for lev, lev_codes in zip(
1036                         self.right.index.levels, self.right.index.codes
1037                     )
1038                 ]
1039             else:
1040                 right_keys = [self.right.index._values]
1041         elif _any(self.right_on):
1042             for k in self.right_on:
1043                 if is_rkey(k):
1044                     right_keys.append(k)
1045                     join_names.append(None)
1046                 else:
1047                     right_keys.append(right._get_label_or_level_values(k))
1048                     join_names.append(k)
1049             if isinstance(self.left.index, MultiIndex):
1050                 left_keys = [
1051                     lev._values.take(lev_codes)
1052                     for lev, lev_codes in zip(
1053                         self.left.index.levels, self.left.index.codes
1054                     )
1055                 ]
1056             else:
1057                 left_keys = [self.left.index._values]
1058 
1059         if left_drop:
1060             self.left = self.left._drop_labels_or_levels(left_drop)
1061 
1062         if right_drop:
1063             self.right = self.right._drop_labels_or_levels(right_drop)
1064 
1065         return left_keys, right_keys, join_names

1067     def _maybe_coerce_merge_keys(self):
1068         # we have valid merges but we may have to further
1069         # coerce these if they are originally incompatible types
1070         #
1071         # for example if these are categorical, but are not dtype_equal
1072         # or if we have object and integer dtypes
1073 
1074         for lk, rk, name in zip(
1075             self.left_join_keys, self.right_join_keys, self.join_names
1076         ):
1077             if (len(lk) and not len(rk)) or (not len(lk) and len(rk)):
1078                 continue
1079 
1080             lk_is_cat = is_categorical_dtype(lk.dtype)
1081             rk_is_cat = is_categorical_dtype(rk.dtype)
1082             lk_is_object = is_object_dtype(lk.dtype)
1083             rk_is_object = is_object_dtype(rk.dtype)
1084 
1085             # if either left or right is a categorical
1086             # then the must match exactly in categories & ordered
1087             if lk_is_cat and rk_is_cat:
1088                 if lk.is_dtype_equal(rk):
1089                     continue
1090 
1091             elif lk_is_cat or rk_is_cat:
1092                 pass
1093 
1094             elif is_dtype_equal(lk.dtype, rk.dtype):
1095                 continue
1096 
1097             msg = (
1098                 f"You are trying to merge on {lk.dtype} and "
1099                 f"{rk.dtype} columns. If you wish to proceed you should use pd.concat"
1100             )
1101 
1102             # if we are numeric, then allow differing
1103             # kinds to proceed, eg. int64 and int8, int and float
1104             # further if we are object, but we infer to
1105             # the same, then proceed
1106             if is_numeric_dtype(lk.dtype) and is_numeric_dtype(rk.dtype):
1107                 if lk.dtype.kind == rk.dtype.kind:
1108                     continue
1109 
1110                 # check whether ints and floats
1111                 elif is_integer_dtype(rk.dtype) and is_float_dtype(lk.dtype):
1112                     if not (lk == lk.astype(rk.dtype))[~np.isnan(lk)].all():
1113                         warnings.warn(
1114                             "You are merging on int and float "
1115                             "columns where the float values "
1116                             "are not equal to their int representation",
1117                             UserWarning,
1118                         )
1119                     continue
1120 
1121                 elif is_float_dtype(rk.dtype) and is_integer_dtype(lk.dtype):
1122                     if not (rk == rk.astype(lk.dtype))[~np.isnan(rk)].all():
1123                         warnings.warn(
1124                             "You are merging on int and float "
1125                             "columns where the float values "
1126                             "are not equal to their int representation",
1127                             UserWarning,
1128                         )
1129                     continue
1130 
1131                 # let's infer and see if we are ok
1132                 elif lib.infer_dtype(lk, skipna=False) == lib.infer_dtype(
1133                     rk, skipna=False
1134                 ):
1135                     continue
1136 
1137             # Check if we are trying to merge on obviously
1138             # incompatible dtypes GH 9780, GH 15800
1139 
1140             # bool values are coerced to object
1141             elif (lk_is_object and is_bool_dtype(rk.dtype)) or (
1142                 is_bool_dtype(lk.dtype) and rk_is_object
1143             ):
1144                 pass
1145 
1146             # object values are allowed to be merged
1147             elif (lk_is_object and is_numeric_dtype(rk.dtype)) or (
1148                 is_numeric_dtype(lk.dtype) and rk_is_object
1149             ):
1150                 inferred_left = lib.infer_dtype(lk, skipna=False)
1151                 inferred_right = lib.infer_dtype(rk, skipna=False)
1152                 bool_types = ["integer", "mixed-integer", "boolean", "empty"]
1153                 string_types = ["string", "unicode", "mixed", "bytes", "empty"]
1154 
1155                 # inferred bool
1156                 if inferred_left in bool_types and inferred_right in bool_types:
1157                     pass
1158 
1159                 # unless we are merging non-string-like with string-like
1160                 elif (
1161                     inferred_left in string_types and inferred_right not in string_types
1162                 ) or (
1163                     inferred_right in string_types and inferred_left not in string_types
1164                 ):
1165                     raise ValueError(msg)
1166 
1167             # datetimelikes must match exactly
1168             elif needs_i8_conversion(lk.dtype) and not needs_i8_conversion(rk.dtype):
1169                 raise ValueError(msg)
1170             elif not needs_i8_conversion(lk.dtype) and needs_i8_conversion(rk.dtype):
1171                 raise ValueError(msg)
1172             elif is_datetime64tz_dtype(lk.dtype) and not is_datetime64tz_dtype(
1173                 rk.dtype
1174             ):
1175                 raise ValueError(msg)
1176             elif not is_datetime64tz_dtype(lk.dtype) and is_datetime64tz_dtype(
1177                 rk.dtype
1178             ):
1179                 raise ValueError(msg)
1180 
1181             elif lk_is_object and rk_is_object:
1182                 continue
1183 
1184             # Houston, we have a problem!
1185             # let's coerce to object if the dtypes aren't
1186             # categorical, otherwise coerce to the category
1187             # dtype. If we coerced categories to object,
1188             # then we would lose type information on some
1189             # columns, and end up trying to merge
1190             # incompatible dtypes. See GH 16900.
1191             if name in self.left.columns:
1192                 typ = lk.categories.dtype if lk_is_cat else object
1193                 self.left = self.left.assign(**{name: self.left[name].astype(typ)})
1194             if name in self.right.columns:
1195                 typ = rk.categories.dtype if rk_is_cat else object
1196                 self.right = self.right.assign(**{name: self.right[name].astype(typ)})

1198     def _validate_specification(self):
1199         # Hm, any way to make this logic less complicated??
1200         if self.on is None and self.left_on is None and self.right_on is None:
1201 
1202             if self.left_index and self.right_index:

1204             elif self.left_index:

1206                     raise MergeError("Must pass right_on or right_index=True")

1871 def _factorize_keys(
1872     lk: ArrayLike, rk: ArrayLike, sort: bool = True, how: str = "inner"
1873 ) -> Tuple[np.array, np.array, int]:
1874     """
1875     Encode left and right keys as enumerated types.
1876 
1877     This is used to get the join indexers to be used when merging DataFrames.
1878 
1879     Parameters
1880     ----------
1881     lk : array-like
1882         Left key.
1883     rk : array-like
1884         Right key.
1885     sort : bool, defaults to True
1886         If True, the encoding is done such that the unique elements in the
1887         keys are sorted.
1888     how : {left, right, outer, inner}, default inner
1889         Type of merge.
1890 
1891     Returns
1892     -------
1893     array
1894         Left (resp. right if called with `key='right'`) labels, as enumerated type.
1895     array
1896         Right (resp. left if called with `key='right'`) labels, as enumerated type.
1897     int
1898         Number of unique elements in union of left and right labels.
1899 
1900     See Also
1901     --------
1902     merge : Merge DataFrame or named Series objects
1903         with a database-style join.
1904     algorithms.factorize : Encode the object as an enumerated type
1905         or categorical variable.
1906 
1907     Examples
1908     --------
1909     >>> lk = np.array(["a", "c", "b"])
1910     >>> rk = np.array(["a", "c"])
1911 
1912     Here, the unique values are `'a', 'b', 'c'`. With the default
1913     `sort=True`, the encoding will be `{0: 'a', 1: 'b', 2: 'c'}`:
1914 
1915     >>> pd.core.reshape.merge._factorize_keys(lk, rk)
1916     (array([0, 2, 1]), array([0, 2]), 3)
1917 
1918     With the `sort=False`, the encoding will correspond to the order
1919     in which the unique elements first appear: `{0: 'a', 1: 'c', 2: 'b'}`:
1920 
1921     >>> pd.core.reshape.merge._factorize_keys(lk, rk, sort=False)
1922     (array([0, 1, 2]), array([0, 1]), 3)
1923     """
1924     # Some pre-processing for non-ndarray lk / rk
1925     lk = extract_array(lk, extract_numpy=True)
1926     rk = extract_array(rk, extract_numpy=True)
1927 
1928     if is_datetime64tz_dtype(lk.dtype) and is_datetime64tz_dtype(rk.dtype):
1929         # Extract the ndarray (UTC-localized) values
1930         # Note: we dont need the dtypes to match, as these can still be compared
1931         lk, _ = lk._values_for_factorize()
1932         rk, _ = rk._values_for_factorize()
1933 
1934     elif (
1935         is_categorical_dtype(lk) and is_categorical_dtype(rk) and is_dtype_equal(lk, rk)
1936     ):
1937         assert isinstance(lk, Categorical)
1938         assert isinstance(rk, Categorical)
1939         if lk.categories.equals(rk.categories):
1940             # if we exactly match in categories, allow us to factorize on codes
1941             rk = rk.codes
1942         else:
1943             # Same categories in different orders -> recode
1944             rk = recode_for_categories(rk.codes, rk.categories, lk.categories)
1945 
1946         lk = ensure_int64(lk.codes)
1947         rk = ensure_int64(rk)
1948 
1949     elif is_extension_array_dtype(lk.dtype) and is_dtype_equal(lk.dtype, rk.dtype):
1950         lk, _ = lk._values_for_factorize()
1951         rk, _ = rk._values_for_factorize()
1952 
1953     if is_integer_dtype(lk) and is_integer_dtype(rk):
1954         # GH#23917 TODO: needs tests for case where lk is integer-dtype
1955         #  and rk is datetime-dtype
1956         klass = libhashtable.Int64Factorizer
1957         lk = ensure_int64(np.asarray(lk))
1958         rk = ensure_int64(np.asarray(rk))
1959 
1960     elif needs_i8_conversion(lk.dtype) and is_dtype_equal(lk.dtype, rk.dtype):
1961         # GH#23917 TODO: Needs tests for non-matching dtypes
1962         klass = libhashtable.Int64Factorizer
1963         lk = ensure_int64(np.asarray(lk, dtype=np.int64))
1964         rk = ensure_int64(np.asarray(rk, dtype=np.int64))
1965 
1966     else:
1967         klass = libhashtable.Factorizer
1968         lk = ensure_object(lk)
1969         rk = ensure_object(rk)
1970 
1971     rizer = klass(max(len(lk), len(rk)))
1972 
1973     llab = rizer.factorize(lk)
1974     rlab = rizer.factorize(rk)
1975 
1976     count = rizer.get_count()
1977 
1978     if sort:
1979         uniques = rizer.uniques.to_array()
1980         llab, rlab = _sort_labels(uniques, llab, rlab)
1981 
1982     # NA group
1983     lmask = llab == -1
1984     lany = lmask.any()
1985     rmask = rlab == -1
1986     rany = rmask.any()
1987 
1988     if lany or rany:
1989         if lany:
1990             np.putmask(llab, lmask, count)
1991         if rany:
1992             np.putmask(rlab, rmask, count)
1993         count += 1
1994 
1995     if how == "right":
1996         return rlab, llab, count
1997     return llab, rlab, count

2003     labels = np.concatenate([left, right])
2004

2012 def _get_join_keys(llab, rlab, shape, sort: bool):
2013 
2014     # how many levels can be done without overflow
2015     pred = lambda i: not is_int64_overflow_possible(shape[:i])
2016     nlev = next(filter(pred, range(len(shape), 0, -1)))
2017 
2018     # get keys for the first `nlev` levels
2019     stride = np.prod(shape[1:nlev], dtype="i8")
2020     lkey = stride * llab[0].astype("i8", subok=False, copy=False)
2021     rkey = stride * rlab[0].astype("i8", subok=False, copy=False)
2022 
2023     for i in range(1, nlev):
2024         with np.errstate(divide="ignore"):
2025             stride //= shape[i]
2026         lkey += llab[i] * stride
2027         rkey += rlab[i] * stride
2028 
2029     if nlev == len(shape):  # all done!
2030         return lkey, rkey
2031 
2032     # densify current keys to avoid overflow
2033     lkey, rkey, count = _factorize_keys(lkey, rkey, sort=sort)
2034 
2035     llab = [lkey] + llab[nlev:]
2036     rlab = [rkey] + rlab[nlev:]
2037     shape = [count] + shape[nlev:]
2038 
2039     return _get_join_keys(llab, rlab, shape, sort)
