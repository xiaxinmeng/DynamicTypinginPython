  50 def _normalized(p):
  51     if p is None:
  52         return None
  53     loc = vistir.compat.Path(p)
  54     try:
  55         loc = loc.resolve()
  56     except OSError:
  57         loc = loc.absolute()
  58     # Recase the path properly on Windows. From https://stackoverflow.com/a/35229734/5043728
  59     if os.name == 'nt':
  60         matches = glob.glob(re.sub(r'([^:/\\])(?=[/\\]|$)', r'[\1]', str(loc)))
  61         path_str = matches and matches[0] or str(loc)
  62     else:
  63         path_str = str(loc)
  64     return normalize_drive(os.path.abspath(path_str))

  70 class _LockFileEncoder(json.JSONEncoder):
  71     """A specilized JSON encoder to convert loaded TOML data into a lock file.
  72 
  73     This adds a few characteristics to the encoder:
  74 
  75     * The JSON is always prettified with indents and spaces.
  76     * TOMLKit's container elements are seamlessly encodable.
  77     * The output is always UTF-8-encoded text, never binary, even on Python 2.
  78     """
  79 
  80     def __init__(self):
  81         super(_LockFileEncoder, self).__init__(
  82             indent=4, separators=(",", ": "), sort_keys=True
  83         )
  84 
  85     def default(self, obj):
  86         if isinstance(obj, vistir.compat.Path):
  87             obj = obj.as_posix()
  88         return super(_LockFileEncoder, self).default(obj)

  90     def encode(self, obj):
  91         content = super(_LockFileEncoder, self).encode(obj)
  92         if not isinstance(content, six.text_type):
  93             content = content.decode("utf-8")
  94         return content

 174     def _build_package_list(self, package_section):
 175         """Returns a list of packages for pip-tools to consume."""
 176         from pipenv.vendor.requirementslib.utils import is_vcs
 177         ps = {}
 178         # TODO: Separate the logic for showing packages from the filters for supplying pip-tools
 179         for k, v in self.parsed_pipfile.get(package_section, {}).items():
 180             # Skip editable VCS deps.
 181             if hasattr(v, "keys"):
 182                 # When a vcs url is gven without editable it only appears as a key
 183                 # Eliminate any vcs, path, or url entries which are not editable
 184                 # Since pip-tools can't do deep resolution on them, even setuptools-installable ones
 185                 if (
 186                     is_vcs(v)
 187                     or is_vcs(k)
 188                     or (is_installable_file(k) or is_installable_file(v))
 189                     or any(
 190                         (
 191                             prefix in v
 192                             and (os.path.isfile(v[prefix]) or is_valid_url(v[prefix]))
 193                         )
 194                         for prefix in ["path", "file"]
 195                     )
 196                 ):
 197                     # If they are editable, do resolve them
 198                     if "editable" not in v:
 199                         # allow wheels to be passed through
 200                         if not (
 201                             hasattr(v, "keys")
 202                             and v.get("path", v.get("file", "")).endswith(".whl")
 203                         ):
 204                             continue
 205                         ps.update({k: v})
 206 
 207                     else:
 208                         ps.update({k: v})
 209                 else:
 210                     ps.update({k: v})
 211             else:
 212                 # Since these entries have no attributes we know they are not editable
 213                 # So we can safely exclude things that need to be editable in order to be resolved
 214                 # First exclude anything that is a vcs entry either in the key or value
 215                 if not (
 216                     any(is_vcs(i) for i in [k, v])
 217                     # Then exclude any installable files that are not directories
 218                     # Because pip-tools can resolve setup.py for example
 219                     or any(is_installable_file(i) for i in [k, v])
 220                     # Then exclude any URLs because they need to be editable also
 221                     # Things that are excluded can only be 'shallow resolved'
 222                     or any(is_valid_url(i) for i in [k, v])
 223                 ):
 224                     ps.update({k: v})
 225         return ps

 280     def get_location_for_virtualenv(self):
 281         # type: () -> str
 282         # If there's no project yet, set location based on config.
 283         if not self.project_directory:
 284             if self.is_venv_in_project():
 285                 return os.path.abspath(".venv")
 286             return str(get_workon_home().joinpath(self.virtualenv_name))
 287 
 288         dot_venv = os.path.join(self.project_directory, ".venv")
 289 
 290         # If there's no .venv in project root, set location based on config.
 291         if not os.path.exists(dot_venv):
 292             if self.is_venv_in_project():
 293                 return dot_venv
 294             return str(get_workon_home().joinpath(self.virtualenv_name))
 295 
 296         # If .venv in project root is a directory, use it.
 297         if os.path.isdir(dot_venv):
 298             return dot_venv
 299 
 300         # Now we assume .venv in project root is a file. Use its content.
 301         with io.open(dot_venv) as f:
 302             name = f.read().strip()
 303 
 304         # If content looks like a path, use it as a relative path.
 305         # Otherwise use directory named after content in WORKON_HOME.
 306         if looks_like_dir(name):
 307             path = vistir.compat.Path(self.project_directory, name)
 308             return path.absolute().as_posix()
 309         return str(get_workon_home().joinpath(name))

 383     @classmethod
 384     def _sanitize(cls, name):
 385         # type: (str) -> Tuple[str, str]
 386         # Replace dangerous characters into '_'. The length of the sanitized
 387         # project name is limited as 42 because of the limit of linux kernel
 388         #
 389         # 42 = 127 - len('/home//.local/share/virtualenvs//bin/python2') - 32 - len('-HASHHASH')
 390         #
 391         #      127 : BINPRM_BUF_SIZE - 1
 392         #       32 : Maximum length of username
 393         #
 394         # References:
 395         #   https://www.gnu.org/software/bash/manual/html_node/Double-Quotes.html
 396         #   http://www.tldp.org/LDP/abs/html/special-chars.html#FIELDREF
 397         #   https://github.com/torvalds/linux/blob/2bfe01ef/include/uapi/linux/binfmts.h#L18
 398         return re.sub(r'[ $`!*@"\\\r\n\t]', "_", name)[0:42]

 400     def _get_virtualenv_hash(self, name):
 401         # type: (str) -> str
 402         """Get the name of the virtualenv adjusted for windows if needed
 403 
 404         Returns (name, encoded_hash)
 405         """
 406 
 407         def get_name(name, location):
 408             name = self._sanitize(name)
 409             hash = hashlib.sha256(location.encode()).digest()[:6]
 410             encoded_hash = base64.urlsafe_b64encode(hash).decode()
 411             return name, encoded_hash[:8]
 412 
 413         clean_name, encoded_hash = get_name(name, self.pipfile_location)
 414         venv_name = "{0}-{1}".format(clean_name, encoded_hash)
 415 
 416         # This should work most of the time for
 417         #   Case-sensitive filesystems,
 418         #   In-project venv
 419         #   "Proper" path casing (on non-case-sensitive filesystems).
 420         if (
 421             not fnmatch.fnmatch("A", "a")
 422             or self.is_venv_in_project()
 423             or get_workon_home().joinpath(venv_name).exists()
 424         ):
 425             return clean_name, encoded_hash
 426 
 427         # Check for different capitalization of the same project.
 428         for path in get_workon_home().iterdir():
 429             if not is_virtual_environment(path):
 430                 continue
 431             try:
 432                 env_name, hash_ = path.name.rsplit("-", 1)
 433             except ValueError:
 434                 continue
 435             if len(hash_) != 8 or env_name.lower() != name.lower():
 436                 continue
 437             return get_name(env_name, self.pipfile_location.replace(name, env_name))
 438 
 439         # Use the default if no matching env exists.
 440         return clean_name, encoded_hash

 536         # type: () -> Union[tomlkit.toml_document.TOMLDocument, TPipfile]
 537         """Parse Pipfile into a TOMLFile and cache it

 573     def _read_pyproject(self):
 574         # type: () -> None
 575         pyproject = self.path_to("pyproject.toml")
 576         if os.path.exists(pyproject):
 577             self._pyproject = toml.load(pyproject)
 578             build_system = self._pyproject.get("build-system", None)
 579             if not os.path.exists(self.path_to("setup.py")):
 580                 if not build_system or not build_system.get("requires"):

 728     def create_pipfile(self, python=None):
 729         """Creates the Pipfile, filled with juicy defaults."""
 730         from .vendor.pip_shims.shims import InstallCommand
 731         # Inherit the pip's index configuration of install command.
 732         command = InstallCommand()
 733         indexes = command.cmd_opts.get_option("--extra-index-url").default
 734         sources = [DEFAULT_SOURCE]
 735         for i, index in enumerate(indexes):
 736             if not index:
 737                 continue
 738 
 739             source_name = "pip_index_{}".format(i)
 740             verify_ssl = index.startswith("https")
 741             sources.append(
 742                 {u"url": index, u"verify_ssl": verify_ssl, u"name": source_name}
 743             )
 744 
 745         data = {
 746             u"source": sources,
 747             # Default packages.
 748             u"packages": {},
 749             u"dev-packages": {},
 750         }
 751         # Default requires.
 752         required_python = python
 753         if not python:
 754             if self.virtualenv_location:
 755                 required_python = self.which("python", self.virtualenv_location)
 756             else:
 757                 required_python = self.which("python")
 758         version = python_version(required_python) or PIPENV_DEFAULT_PYTHON_VERSION
 759         if version and len(version) >= 3:
 760             data[u"requires"] = {"python_version": version[: len("2.7")]}
 761         self.write_toml(data)

 829     def write_toml(self, data, path=None):
 830         """Writes the given data structure out as TOML."""
 831         if path is None:
 832             path = self.pipfile_location
 833         data = convert_toml_outline_tables(data)
 834         try:
 835             formatted_data = tomlkit.dumps(data).rstrip()
 836         except Exception:
 837             document = tomlkit.document()
 838             for section in ("packages", "dev-packages"):
 839                 document[section] = tomlkit.table()
 840                 # Convert things to inline tables  fancy :)
 841                 for package in data.get(section, {}):
 842                     if hasattr(data[section][package], "keys"):
 843                         table = tomlkit.inline_table()
 844                         table.update(data[section][package])
 845                         document[section][package] = table
 846                     else:
 847                         document[section][package] = tomlkit.string(data[section][package])
 848             formatted_data = tomlkit.dumps(document).rstrip()
 849 
 850         if (
 851             vistir.compat.Path(path).absolute()
 852             == vistir.compat.Path(self.pipfile_location).absolute()
 853         ):
 854             newlines = self._pipfile_newlines
 855         else:
 856             newlines = DEFAULT_NEWLINES
 857         formatted_data = cleanup_toml(formatted_data)
 858         with io.open(path, "w", newline=newlines) as f:
 859             f.write(formatted_data)
 860         # pipfile is mutated!
 861         self.clear_pipfile_cache()

 863     def write_lockfile(self, content):
 864         """Write out the lockfile.
 865         """
 866         s = self._lockfile_encoder.encode(content)
 867         open_kwargs = {"newline": self._lockfile_newlines, "encoding": "utf-8"}
 868         with vistir.contextmanagers.atomic_open_for_write(

 906         Given a source, find it.

 919     def get_source(self, name=None, url=None, refresh=False):
 920         from .utils import is_url_equal
 921 
 922         def find_source(sources, name=None, url=None):
 923             source = None
 924             if name:
 925                 source = next(iter(
 926                     s for s in sources if "name" in s and s["name"] == name
 927                 ), None)
 928             elif url:
 929                 source = next(iter(
 930                     s for s in sources
 931                     if "url" in s and is_url_equal(url, s.get("url", ""))
 932                 ), None)
 933             if source is not None:
 934                 return source
 935 
 936         sources = (self.sources, self.pipfile_sources)
 937         if refresh:
 938             self.clear_pipfile_cache()
 939             sources = reversed(sources)
 940         found = next(
 941             iter(find_source(source, name=name, url=url) for source in sources), None
 942         )
 943         target = next(iter(t for t in (name, url) if t is not None))
 944         if found is None:
 945             raise SourceNotFound(target)
 946         return found

1085     def proper_case_section(self, section):
1086         """Verify proper casing is retrieved, when available, for each
1087         dependency in the section.
1088         """
1089         # Casing for section.
1090         changed_values = False
1091         unknown_names = [k for k in section.keys() if k not in set(self.proper_names)]
1092         # Replace each package with proper casing.
1093         for dep in unknown_names:
1094             try:
1095                 # Get new casing for package name.
1096                 new_casing = proper_case(dep)
1097             except IOError:
1098                 # Unable to normalize package name.
1099                 continue
1100 
1101             if new_casing != dep:
1102                 changed_values = True
1103                 self.register_proper_name(new_casing)
1104                 # Replace old value with new value.
1105                 old_value = section[dep]
1106                 section[new_casing] = old_value
1107                 del section[dep]
1108         # Return whether or not values have been changed.
1109         return changed_values
