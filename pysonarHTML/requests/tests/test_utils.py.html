 707     'url, expected, override', (
 708             ('http://192.168.0.1:5000/', True, None),
 709             ('http://192.168.0.1/', True, None),

 713             ('http://172.16.1.22/', False, None),

 715             ('http://google.com:5000/v1.0/', False, None),
 716             ('http://mylocalhostname:5000/v1.0/', True, '<local>'),
 717             ('http://192.168.0.1/', False, ''),
 718     ))
 719 def test_should_bypass_proxies_win_registry(url, expected, override,
 720                                             monkeypatch):
 721     """Tests for function should_bypass_proxies to check if proxy
 722     can be bypassed or not with Windows registry settings
 723     """
 724     if override is None:
 725         override = '192.168.*;127.0.0.1;localhost.localdomain;172.16.1.1'
 726     if compat.is_py3:
 727         import winreg
 728     else:
 729         import _winreg as winreg
 730 
 731     class RegHandle:
 732         def Close(self):
 733             pass
 734 
 735     ie_settings = RegHandle()
 736     proxyEnableValues = deque([1, "1"])
 737 
 738     def OpenKey(key, subkey):
 739         return ie_settings
 740 
 741     def QueryValueEx(key, value_name):
 742         if key is ie_settings:
 743             if value_name == 'ProxyEnable':
 744                 # this could be a string (REG_SZ) or a 32-bit number (REG_DWORD)
 745                 proxyEnableValues.rotate()
 746                 return [proxyEnableValues[0]]
 747             elif value_name == 'ProxyOverride':
 748                 return [override]
 749 
 750     monkeypatch.setenv('http_proxy', '')
 751     monkeypatch.setenv('https_proxy', '')
 752     monkeypatch.setenv('ftp_proxy', '')
 753     monkeypatch.setenv('no_proxy', '')
 754     monkeypatch.setenv('NO_PROXY', '')
 755     monkeypatch.setattr(winreg, 'OpenKey', OpenKey)
 756     monkeypatch.setattr(winreg, 'QueryValueEx', QueryValueEx)
 757     assert should_bypass_proxies(url, None) == expected
