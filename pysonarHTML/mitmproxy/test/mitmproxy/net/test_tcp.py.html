  44 class HangHandler(tcp.BaseHandler):
  45 
  46     def handle(self):
  47         # Hang as long as the client connection is alive
  48         while True:
  49             try:
  50                 self.connection.setblocking(0)
  51                 ret = self.connection.recv(1)
  52                 # Client connection is dead...
  53                 if ret == "" or ret == b"":
  54                     return
  55             except socket.error:
  56                 pass
  57             except SSL.WantReadError:
  58                 pass
  59             except Exception:
  60                 return
  61             time.sleep(0.1)

  99     class handler(tcp.BaseHandler):
 100 
 101         def handle(self):
 102             # We may get an ipv4-mapped ipv6 address here, e.g. ::ffff:127.0.0.1.
 103             # Those still appear as "127.0.0.1" in the table, so we need to strip the prefix.
 104             peername = self.connection.getpeername()
 105             address = re.sub(r"^::ffff:(?=\d+.\d+.\d+.\d+$)", "", peername[0])
 106             port = peername[1]
 107 
 108             self.wfile.write(str((address, port)).encode())
 109             self.wfile.flush()
 110 
 111     def test_bind(self):
 112         """ Test to bind to a given random port. Try again if the random port turned out to be blocked. """
 113         for i in range(20):
 114             random_port = random.randrange(1024, 65535)
 115             try:
 116                 c = tcp.TCPClient(
 117                     ("127.0.0.1", self.port), source_address=(
 118                         "127.0.0.1", random_port))
 119                 with c.connect():
 120                     assert c.rfile.readline() == str(("127.0.0.1", random_port)).encode()
 121                     return
 122             except exceptions.TcpException:  # port probably already in use
 123                 pass
