   9 def _test_passthrough_attr(message, attr):
  10     assert getattr(message, attr) == getattr(message.data, attr)
  11     setattr(message, attr, b"foo")
  12     assert getattr(message.data, attr) == b"foo"
  13

  15 def _test_decoded_attr(message, attr):
  16     assert getattr(message, attr) == getattr(message.data, attr).decode("utf8")
  17     # Set str, get raw bytes
  18     setattr(message, attr, "foo")
  19     assert getattr(message.data, attr) == b"foo"
  20     # Set raw bytes, get decoded
  21     setattr(message.data, attr, b"BAR")  # use uppercase so that we can also cover request.method
  22     assert getattr(message, attr) == "BAR"
  23     # Set bytes, get raw bytes
  24     setattr(message, attr, b"baz")
  25     assert getattr(message.data, attr) == b"baz"
  26 
  27     # Set UTF8
  28     setattr(message, attr, "Non-Autoris√©")
  29     assert getattr(message.data, attr) == b"Non-Autoris\xc3\xa9"
  30     # Don't fail on garbage
  31     setattr(message.data, attr, b"FOO\xBF\x00BAR")
  32     assert getattr(message, attr).startswith("FOO")
  33     assert getattr(message, attr).endswith("BAR")
  34     # foo.bar = foo.bar should not cause any side effects.
  35     d = getattr(message, attr)
  36     setattr(message, attr, d)
  37     assert getattr(message.data, attr) == b"FOO\xBF\x00BAR"
