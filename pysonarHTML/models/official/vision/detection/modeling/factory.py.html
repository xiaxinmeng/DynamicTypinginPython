  15 """Factory to build detection model."""
  16 
  17 
  18 from official.vision.detection.modeling import maskrcnn_model
  19 from official.vision.detection.modeling import retinanet_model
  20 from official.vision.detection.modeling import shapemask_model
  21 
  22 
  23 def model_generator(params):
  24   """Model function generator."""
  25   if params.type == 'retinanet':
  26     model_fn = retinanet_model.RetinanetModel(params)
  27   elif params.type == 'mask_rcnn':
  28     model_fn = maskrcnn_model.MaskrcnnModel(params)
  29   elif params.type == 'shapemask':
  30     model_fn = shapemask_model.ShapeMaskModel(params)
  31   else:
  32     raise ValueError('Model %s is not supported.'% params.type)
  33 
  34   return model_fn
