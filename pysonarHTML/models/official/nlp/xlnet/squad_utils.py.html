  79   qid_to_has_ans = {}

  82       for qa in p["qas"]:

  87 def get_raw_scores(dataset, preds):
  88   """Gets exact scores and f1 scores."""
  89   exact_scores = {}
  90   f1_scores = {}
  91   for article in dataset:
  92     for p in article["paragraphs"]:
  93       for qa in p["qas"]:
  94         qid = qa["id"]
  95         gold_answers = [
  96             a["text"] for a in qa["answers"] if normalize_answer(a["text"])
  97         ]
  98         if not gold_answers:
  99           # For unanswerable questions, only correct answer is empty string
 100           gold_answers = [""]
 101         if qid not in preds:
 102           print("Missing prediction for %s" % qid)
 103           continue
 104         a_pred = preds[qid]
 105         # Take max over all gold answers
 106         exact_scores[qid] = max(compute_exact(a, a_pred) for a in gold_answers)
 107         f1_scores[qid] = max(compute_f1(a, a_pred) for a in gold_answers)
 108   return exact_scores, f1_scores

 111 def normalize_answer(s):
 112   """Lower text and remove punctuation, articles and extra whitespace."""
 113 
 114   def remove_articles(text):
 115     regex = re.compile(r"\b(a|an|the)\b", re.UNICODE)
 116     return re.sub(regex, " ", text)
 117 
 118   def white_space_fix(text):
 119     return " ".join(text.split())
 120 
 121   def remove_punc(text):
 122     exclude = set(string.punctuation)
 123     return "".join(ch for ch in text if ch not in exclude)
 124 
 125   def lower(text):
 126     return text.lower()
 127 
 128   return white_space_fix(remove_articles(remove_punc(lower(s))))

 141 def compute_f1(a_gold, a_pred):
 142   """Computes f1 score."""
 143   gold_toks = get_tokens(a_gold)
 144   pred_toks = get_tokens(a_pred)
 145   common = collections.Counter(gold_toks) & collections.Counter(pred_toks)
 146   num_same = sum(common.values())
 147   # pylint: disable=g-explicit-length-test
 148   if len(gold_toks) == 0 or len(pred_toks) == 0:
 149     # If either is no-answer, then F1 is 1 if they agree, 0 otherwise
 150     return int(gold_toks == pred_toks)
 151   if num_same == 0:
 152     return 0
 153   precision = 1.0 * num_same / len(pred_toks)
 154   recall = 1.0 * num_same / len(gold_toks)
 155   f1 = (2 * precision * recall) / (precision + recall)
 156   return f1

 159 def find_best_thresh(preds, scores, na_probs, qid_to_has_ans):
 160   """Finds best threshold."""
 161   num_no_ans = sum(1 for k in qid_to_has_ans if not qid_to_has_ans[k])
 162   cur_score = num_no_ans
 163   best_score = cur_score
 164   best_thresh = 0.0
 165   qid_list = sorted(na_probs, key=lambda k: na_probs[k])
 166   for qid in qid_list:
 167     if qid not in scores:
 168       continue
 169     if qid_to_has_ans[qid]:
 170       diff = scores[qid]
 171     else:
 172       if preds[qid]:
 173         diff = -1
 174       else:
 175         diff = 0
 176     cur_score += diff
 177     if cur_score > best_score:
 178       best_score = cur_score
 179       best_thresh = na_probs[qid]
 180 
 181   has_ans_score, has_ans_cnt = 0, 0
 182   for qid in qid_list:
 183     if not qid_to_has_ans[qid]:
 184       continue
 185     has_ans_cnt += 1
 186 
 187     if qid not in scores:
 188       continue
 189     has_ans_score += scores[qid]
 190 
 191   return 100.0 * best_score / len(
 192       scores), best_thresh, 1.0 * has_ans_score / has_ans_cnt

 195 def find_all_best_thresh(main_eval, preds, exact_raw, f1_raw, na_probs,
 196                          qid_to_has_ans):
 197   """Finds all best threshold."""
 198   best_exact, exact_thresh, has_ans_exact = find_best_thresh(
 199       preds, exact_raw, na_probs, qid_to_has_ans)
 200   best_f1, f1_thresh, has_ans_f1 = find_best_thresh(preds, f1_raw, na_probs,
 201                                                     qid_to_has_ans)
 202   main_eval["best_exact"] = best_exact
 203   main_eval["best_exact_thresh"] = exact_thresh
 204   main_eval["best_f1"] = best_f1
 205   main_eval["best_f1_thresh"] = f1_thresh
 206   main_eval["has_ans_exact"] = has_ans_exact
 207   main_eval["has_ans_f1"] = has_ans_f1

 224 def _compute_softmax(scores):
 225   """Computes softmax probability over raw logits."""
 226   if not scores:
 227     return []
 228 
 229   max_score = None
 230   for score in scores:
 231     if max_score is None or score > max_score:
 232       max_score = score
 233 
 234   exp_scores = []
 235   total_sum = 0.0
 236   for score in scores:
 237     x = math.exp(score - max_score)
 238     exp_scores.append(x)
 239     total_sum += x
 240 
 241   probs = []
 242   for score in exp_scores:
 243     probs.append(score / total_sum)
 244   return probs

 247 class SquadExample(object):
 248   """A single training/test example for simple sequence classification.
 249 
 250      For examples without an answer, the start and end position are -1.
 251   """
 252 
 253   def __init__(self,
 254                qas_id,
 255                question_text,
 256                paragraph_text,
 257                orig_answer_text=None,
 258                start_position=None,
 259                is_impossible=False):
 260     self.qas_id = qas_id
 261     self.question_text = question_text
 262     self.paragraph_text = paragraph_text
 263     self.orig_answer_text = orig_answer_text
 264     self.start_position = start_position
 265     self.is_impossible = is_impossible
 266 
 267   def __str__(self):
 268     return self.__repr__()
 269 
 270   def __repr__(self):
 271     s = ""
 272     s += "qas_id: %s" % (preprocess_utils.printable_text(self.qas_id))
 273     s += ", question_text: %s" % (
 274         preprocess_utils.printable_text(self.question_text))
 275     s += ", paragraph_text: [%s]" % (" ".join(self.paragraph_text))
 276     if self.start_position:
 277       s += ", start_position: %d" % (self.start_position)
 278     if self.start_position:
 279       s += ", is_impossible: %r" % (self.is_impossible)
 280     return s

 283 def write_predictions(all_examples, all_features, all_results, n_best_size,
 284                       max_answer_length, output_prediction_file,
 285                       output_nbest_file, output_null_log_odds_file, orig_data,
 286                       start_n_top, end_n_top):
 287   """Writes final predictions to the json file and log-odds of null if needed."""
 288   logging.info("Writing predictions to: %s", (output_prediction_file))
 289 
 290   example_index_to_features = collections.defaultdict(list)
 291   for feature in all_features:
 292     example_index_to_features[feature.example_index].append(feature)
 293 
 294   unique_id_to_result = {}
 295   for result in all_results:
 296     unique_id_to_result[result.unique_id] = result
 297 
 298   all_predictions = collections.OrderedDict()
 299   all_nbest_json = collections.OrderedDict()
 300   scores_diff_json = collections.OrderedDict()
 301 
 302   for (example_index, example) in enumerate(all_examples):
 303     features = example_index_to_features[example_index]
 304 
 305     prelim_predictions = []
 306     # keep track of the minimum score of null start+end of position 0
 307     score_null = 1000000  # large and positive
 308 
 309     for (feature_index, feature) in enumerate(features):
 310       result = unique_id_to_result[feature.unique_id]
 311 
 312       cur_null_score = result.cls_logits
 313 
 314       # if we could have irrelevant answers, get the min score of irrelevant
 315       score_null = min(score_null, cur_null_score)
 316 
 317       for i in range(start_n_top):
 318         for j in range(end_n_top):
 319           start_log_prob = result.start_top_log_probs[i]
 320           start_index = result.start_top_index[i]
 321 
 322           j_index = i * end_n_top + j
 323 
 324           end_log_prob = result.end_top_log_probs[j_index]
 325           end_index = result.end_top_index[j_index]
 326 
 327           # We could hypothetically create invalid predictions, e.g., predict
 328           # that the start of the span is in the question. We throw out all
 329           # invalid predictions.
 330           if start_index >= feature.paragraph_len - 1:
 331             continue
 332           if end_index >= feature.paragraph_len - 1:
 333             continue
 334 
 335           if not feature.token_is_max_context.get(start_index, False):
 336             continue
 337           if end_index < start_index:
 338             continue
 339           length = end_index - start_index + 1
 340           if length > max_answer_length:
 341             continue
 342 
 343           prelim_predictions.append(
 344               _PrelimPrediction(
 345                   feature_index=feature_index,
 346                   start_index=start_index,
 347                   end_index=end_index,
 348                   start_log_prob=start_log_prob,
 349                   end_log_prob=end_log_prob))
 350 
 351     prelim_predictions = sorted(
 352         prelim_predictions,
 353         key=lambda x: (x.start_log_prob + x.end_log_prob),
 354         reverse=True)
 355 
 356     seen_predictions = {}
 357     nbest = []
 358     for pred in prelim_predictions:
 359       if len(nbest) >= n_best_size:
 360         break
 361       feature = features[pred.feature_index]
 362 
 363       tok_start_to_orig_index = feature.tok_start_to_orig_index
 364       tok_end_to_orig_index = feature.tok_end_to_orig_index
 365       start_orig_pos = tok_start_to_orig_index[pred.start_index]
 366       end_orig_pos = tok_end_to_orig_index[pred.end_index]
 367 
 368       paragraph_text = example.paragraph_text
 369       final_text = paragraph_text[start_orig_pos:end_orig_pos + 1].strip()
 370 
 371       if final_text in seen_predictions:
 372         continue
 373 
 374       seen_predictions[final_text] = True
 375 
 376       nbest.append(
 377           _NbestPrediction(
 378               text=final_text,
 379               start_log_prob=pred.start_log_prob,
 380               end_log_prob=pred.end_log_prob))
 381 
 382     # In very rare edge cases we could have no valid predictions. So we
 383     # just create a nonce prediction in this case to avoid failure.
 384     if not nbest:
 385       nbest.append(
 386           _NbestPrediction(text="", start_log_prob=-1e6, end_log_prob=-1e6))
 387 
 388     total_scores = []
 389     best_non_null_entry = None
 390     for entry in nbest:
 391       total_scores.append(entry.start_log_prob + entry.end_log_prob)
 392       if not best_non_null_entry:
 393         best_non_null_entry = entry
 394 
 395     probs = _compute_softmax(total_scores)
 396 
 397     nbest_json = []
 398     for (i, entry) in enumerate(nbest):
 399       output = collections.OrderedDict()
 400       output["text"] = entry.text
 401       output["probability"] = probs[i]
 402       output["start_log_prob"] = entry.start_log_prob
 403       output["end_log_prob"] = entry.end_log_prob
 404       nbest_json.append(output)
 405 
 406     assert len(nbest_json) >= 1
 407     assert best_non_null_entry is not None
 408 
 409     score_diff = score_null
 410     scores_diff_json[example.qas_id] = score_diff
 411 
 412     all_predictions[example.qas_id] = best_non_null_entry.text
 413 
 414     all_nbest_json[example.qas_id] = nbest_json
 415 
 416   with tf.io.gfile.GFile(output_prediction_file, "w") as writer:
 417     writer.write(json.dumps(all_predictions, indent=4) + "\n")
 418 
 419   with tf.io.gfile.GFile(output_nbest_file, "w") as writer:
 420     writer.write(json.dumps(all_nbest_json, indent=4) + "\n")
 421 
 422   with tf.io.gfile.GFile(output_null_log_odds_file, "w") as writer:
 423     writer.write(json.dumps(scores_diff_json, indent=4) + "\n")
 424 
 425   qid_to_has_ans = make_qid_to_has_ans(orig_data)
 426   exact_raw, f1_raw = get_raw_scores(orig_data, all_predictions)
 427   out_eval = {}
 428 
 429   find_all_best_thresh(out_eval, all_predictions, exact_raw, f1_raw,
 430                        scores_diff_json, qid_to_has_ans)
 431 
 432   return out_eval

 435 def read_squad_examples(input_file, is_training):
 436   """Reads a SQuAD json file into a list of SquadExample."""
 437   with tf.io.gfile.GFile(input_file, "r") as reader:
 438     input_data = json.load(reader)["data"]
 439 
 440   examples = []
 441   for entry in input_data:
 442     for paragraph in entry["paragraphs"]:
 443       paragraph_text = paragraph["context"]
 444 
 445       for qa in paragraph["qas"]:
 446         qas_id = qa["id"]
 447         question_text = qa["question"]
 448         start_position = None
 449         orig_answer_text = None
 450         is_impossible = False
 451 
 452         if is_training:
 453           is_impossible = qa["is_impossible"]
 454           if (len(qa["answers"]) != 1) and (not is_impossible):
 455             raise ValueError(
 456                 "For training, each question should have exactly 1 answer.")
 457           if not is_impossible:
 458             answer = qa["answers"][0]
 459             orig_answer_text = answer["text"]
 460             start_position = answer["answer_start"]
 461           else:
 462             start_position = -1
 463             orig_answer_text = ""
 464 
 465         example = SquadExample(
 466             qas_id=qas_id,
 467             question_text=question_text,
 468             paragraph_text=paragraph_text,
 469             orig_answer_text=orig_answer_text,
 470             start_position=start_position,
 471             is_impossible=is_impossible)
 472         examples.append(example)
 473 
 474   return examples

 478 def _convert_index(index, pos, M=None, is_start=True):
 479   """Converts index."""
 480   if index[pos] is not None:
 481     return index[pos]
 482   N = len(index)
 483   rear = pos
 484   while rear < N - 1 and index[rear] is None:
 485     rear += 1
 486   front = pos
 487   while front > 0 and index[front] is None:
 488     front -= 1
 489   assert index[front] is not None or index[rear] is not None
 490   if index[front] is None:
 491     if index[rear] >= 1:
 492       if is_start:
 493         return 0
 494       else:
 495         return index[rear] - 1
 496     return index[rear]
 497   if index[rear] is None:
 498     if M is not None and index[front] < M - 1:
 499       if is_start:
 500         return index[front] + 1
 501       else:
 502         return M - 1
 503     return index[front]
 504   if is_start:
 505     if index[rear] > index[front] + 1:
 506       return index[front] + 1
 507     else:
 508       return index[rear]
 509   else:
 510     if index[rear] > index[front] + 1:
 511       return index[rear] - 1
 512     else:
 513       return index[front]

 516 def convert_examples_to_features(examples, sp_model, max_seq_length, doc_stride,
 517                                  max_query_length, is_training, output_fn,
 518                                  uncased):
 519   """Loads a data file into a list of `InputBatch`s."""
 520 
 521   cnt_pos, cnt_neg = 0, 0
 522   unique_id = 1000000000
 523   max_N, max_M = 1024, 1024
 524   f = np.zeros((max_N, max_M), dtype=np.float32)
 525 
 526   for (example_index, example) in enumerate(examples):
 527     # pylint: disable=logging-format-interpolation
 528     if example_index % 100 == 0:
 529       logging.info("Converting {}/{} pos {} neg {}".format(
 530           example_index, len(examples), cnt_pos, cnt_neg))
 531 
 532     query_tokens = preprocess_utils.encode_ids(
 533         sp_model,
 534         preprocess_utils.preprocess_text(example.question_text, lower=uncased))
 535 
 536     if len(query_tokens) > max_query_length:
 537       query_tokens = query_tokens[0:max_query_length]
 538 
 539     paragraph_text = example.paragraph_text
 540     para_tokens = preprocess_utils.encode_pieces(
 541         sp_model,
 542         preprocess_utils.preprocess_text(example.paragraph_text, lower=uncased))
 543 
 544     chartok_to_tok_index = []
 545     tok_start_to_chartok_index = []
 546     tok_end_to_chartok_index = []
 547     char_cnt = 0
 548     for i, token in enumerate(para_tokens):
 549       chartok_to_tok_index.extend([i] * len(token))
 550       tok_start_to_chartok_index.append(char_cnt)
 551       char_cnt += len(token)
 552       tok_end_to_chartok_index.append(char_cnt - 1)
 553 
 554     tok_cat_text = "".join(para_tokens).replace(SPIECE_UNDERLINE, " ")
 555     N, M = len(paragraph_text), len(tok_cat_text)
 556 
 557     if N > max_N or M > max_M:
 558       max_N = max(N, max_N)
 559       max_M = max(M, max_M)
 560       f = np.zeros((max_N, max_M), dtype=np.float32)
 561       gc.collect()
 562 
 563     g = {}
 564 
 565     # pylint: disable=cell-var-from-loop
 566     def _lcs_match(max_dist):
 567       """LCS match."""
 568       f.fill(0)
 569       g.clear()
 570 
 571       ### longest common sub sequence
 572       # f[i, j] = max(f[i - 1, j], f[i, j - 1], f[i - 1, j - 1] + match(i, j))
 573       for i in range(N):
 574 
 575         # note(zhiliny):
 576         # unlike standard LCS, this is specifically optimized for the setting
 577         # because the mismatch between sentence pieces and original text will
 578         # be small
 579         for j in range(i - max_dist, i + max_dist):
 580           if j >= M or j < 0:
 581             continue
 582 
 583           if i > 0:
 584             g[(i, j)] = 0
 585             f[i, j] = f[i - 1, j]
 586 
 587           if j > 0 and f[i, j - 1] > f[i, j]:
 588             g[(i, j)] = 1
 589             f[i, j] = f[i, j - 1]
 590 
 591           f_prev = f[i - 1, j - 1] if i > 0 and j > 0 else 0
 592           if (preprocess_utils.preprocess_text(
 593               paragraph_text[i], lower=uncased,
 594               remove_space=False) == tok_cat_text[j] and f_prev + 1 > f[i, j]):
 595             g[(i, j)] = 2
 596             f[i, j] = f_prev + 1
 597 
 598     max_dist = abs(N - M) + 5
 599     for _ in range(2):
 600       _lcs_match(max_dist)
 601       if f[N - 1, M - 1] > 0.8 * N:
 602         break
 603       max_dist *= 2
 604 
 605     orig_to_chartok_index = [None] * N
 606     chartok_to_orig_index = [None] * M
 607     i, j = N - 1, M - 1
 608     while i >= 0 and j >= 0:
 609       if (i, j) not in g:
 610         break
 611       if g[(i, j)] == 2:
 612         orig_to_chartok_index[i] = j
 613         chartok_to_orig_index[j] = i
 614         i, j = i - 1, j - 1
 615       elif g[(i, j)] == 1:
 616         j = j - 1
 617       else:
 618         i = i - 1
 619 
 620     if all(
 621         v is None for v in orig_to_chartok_index) or f[N - 1, M - 1] < 0.8 * N:
 622       print("MISMATCH DETECTED!")
 623       continue
 624 
 625     tok_start_to_orig_index = []
 626     tok_end_to_orig_index = []
 627     for i in range(len(para_tokens)):
 628       start_chartok_pos = tok_start_to_chartok_index[i]
 629       end_chartok_pos = tok_end_to_chartok_index[i]
 630       start_orig_pos = _convert_index(
 631           chartok_to_orig_index, start_chartok_pos, N, is_start=True)
 632       end_orig_pos = _convert_index(
 633           chartok_to_orig_index, end_chartok_pos, N, is_start=False)
 634 
 635       tok_start_to_orig_index.append(start_orig_pos)
 636       tok_end_to_orig_index.append(end_orig_pos)
 637 
 638     if not is_training:
 639       tok_start_position = tok_end_position = None
 640 
 641     if is_training and example.is_impossible:
 642       tok_start_position = -1
 643       tok_end_position = -1
 644 
 645     if is_training and not example.is_impossible:
 646       start_position = example.start_position
 647       end_position = start_position + len(example.orig_answer_text) - 1
 648 
 649       start_chartok_pos = _convert_index(
 650           orig_to_chartok_index, start_position, is_start=True)
 651       tok_start_position = chartok_to_tok_index[start_chartok_pos]
 652 
 653       end_chartok_pos = _convert_index(
 654           orig_to_chartok_index, end_position, is_start=False)
 655       tok_end_position = chartok_to_tok_index[end_chartok_pos]
 656       assert tok_start_position <= tok_end_position
 657 
 658     def _piece_to_id(x):
 659       if six.PY2 and isinstance(x, unicode):
 660         x = x.encode("utf-8")
 661       return sp_model.PieceToId(x)
 662 
 663     all_doc_tokens = list(map(_piece_to_id, para_tokens))
 664 
 665     # The -3 accounts for [CLS], [SEP] and [SEP]
 666     max_tokens_for_doc = max_seq_length - len(query_tokens) - 3
 667 
 668     # We can have documents that are longer than the maximum sequence length.
 669     # To deal with this we do a sliding window approach, where we take chunks
 670     # of the up to our max length with a stride of `doc_stride`.
 671     _DocSpan = collections.namedtuple(  # pylint: disable=invalid-name
 672         "DocSpan", ["start", "length"])
 673     doc_spans = []
 674     start_offset = 0
 675     while start_offset < len(all_doc_tokens):
 676       length = len(all_doc_tokens) - start_offset
 677       if length > max_tokens_for_doc:
 678         length = max_tokens_for_doc
 679       doc_spans.append(_DocSpan(start=start_offset, length=length))
 680       if start_offset + length == len(all_doc_tokens):
 681         break
 682       start_offset += min(length, doc_stride)
 683 
 684     for (doc_span_index, doc_span) in enumerate(doc_spans):
 685       tokens = []
 686       token_is_max_context = {}
 687       segment_ids = []
 688       p_mask = []
 689 
 690       cur_tok_start_to_orig_index = []
 691       cur_tok_end_to_orig_index = []
 692 
 693       for i in range(doc_span.length):
 694         split_token_index = doc_span.start + i
 695 
 696         cur_tok_start_to_orig_index.append(
 697             tok_start_to_orig_index[split_token_index])
 698         cur_tok_end_to_orig_index.append(
 699             tok_end_to_orig_index[split_token_index])
 700 
 701         is_max_context = _check_is_max_context(doc_spans, doc_span_index,
 702                                                split_token_index)
 703         token_is_max_context[len(tokens)] = is_max_context
 704         tokens.append(all_doc_tokens[split_token_index])
 705         segment_ids.append(data_utils.SEG_ID_P)
 706         p_mask.append(0)
 707 
 708       paragraph_len = len(tokens)
 709 
 710       tokens.append(data_utils.SEP_ID)
 711       segment_ids.append(data_utils.SEG_ID_P)
 712       p_mask.append(1)
 713 
 714       # note(zhiliny): we put P before Q
 715       # because during pretraining, B is always shorter than A
 716       for token in query_tokens:
 717         tokens.append(token)
 718         segment_ids.append(data_utils.SEG_ID_Q)
 719         p_mask.append(1)
 720       tokens.append(data_utils.SEP_ID)
 721       segment_ids.append(data_utils.SEG_ID_Q)
 722       p_mask.append(1)
 723 
 724       cls_index = len(segment_ids)
 725       tokens.append(data_utils.CLS_ID)
 726       segment_ids.append(data_utils.SEG_ID_CLS)
 727       p_mask.append(0)
 728 
 729       input_ids = tokens
 730 
 731       # The mask has 0 for real tokens and 1 for padding tokens. Only real
 732       # tokens are attended to.
 733       input_mask = [0] * len(input_ids)
 734 
 735       # Zero-pad up to the sequence length.
 736       while len(input_ids) < max_seq_length:
 737         input_ids.append(0)
 738         input_mask.append(1)
 739         segment_ids.append(data_utils.SEG_ID_PAD)
 740         p_mask.append(1)
 741 
 742       assert len(input_ids) == max_seq_length
 743       assert len(input_mask) == max_seq_length
 744       assert len(segment_ids) == max_seq_length
 745       assert len(p_mask) == max_seq_length
 746 
 747       span_is_impossible = example.is_impossible
 748       start_position = None
 749       end_position = None
 750       if is_training and not span_is_impossible:
 751         # For training, if our document chunk does not contain an annotation
 752         # we throw it out, since there is nothing to predict.
 753         doc_start = doc_span.start
 754         doc_end = doc_span.start + doc_span.length - 1
 755         out_of_span = False
 756         if not (tok_start_position >= doc_start and
 757                 tok_end_position <= doc_end):
 758           out_of_span = True
 759         if out_of_span:
 760           # continue
 761           start_position = 0
 762           end_position = 0
 763           span_is_impossible = True
 764         else:
 765           # note: we put P before Q, so doc_offset should be zero.
 766           # doc_offset = len(query_tokens) + 2
 767           doc_offset = 0
 768           start_position = tok_start_position - doc_start + doc_offset
 769           end_position = tok_end_position - doc_start + doc_offset
 770 
 771       if is_training and span_is_impossible:
 772         start_position = cls_index
 773         end_position = cls_index
 774 
 775       if example_index < 20:
 776         logging.info("*** Example ***")
 777         logging.info("unique_id: %s", unique_id)
 778         logging.info("example_index: %s", example_index)
 779         logging.info("doc_span_index: %s", doc_span_index)
 780         logging.info("tok_start_to_orig_index: %s",
 781                      " ".join([str(x) for x in cur_tok_start_to_orig_index]))
 782         logging.info("tok_end_to_orig_index: %s",
 783                      " ".join([str(x) for x in cur_tok_end_to_orig_index]))
 784         logging.info(
 785             "token_is_max_context: %s", " ".join([
 786                 "%d:%s" % (x, y)
 787                 for (x, y) in six.iteritems(token_is_max_context)
 788             ]))
 789         logging.info("input_ids: %s", " ".join([str(x) for x in input_ids]))
 790         logging.info("input_mask: %s", " ".join([str(x) for x in input_mask]))
 791         logging.info("segment_ids: %s", " ".join([str(x) for x in segment_ids]))
 792 
 793         if is_training and span_is_impossible:
 794           logging.info("impossible example span")
 795 
 796         if is_training and not span_is_impossible:
 797           pieces = [
 798               sp_model.IdToPiece(token)
 799               for token in tokens[start_position:(end_position + 1)]
 800           ]
 801           answer_text = sp_model.DecodePieces(pieces)
 802           logging.info("start_position: %d", start_position)
 803           logging.info("end_position: %d", end_position)
 804           logging.info("answer: %s",
 805                        preprocess_utils.printable_text(answer_text))
 806 
 807           # With multi processing, the example_index is actually the index
 808           # within the current process therefore we use example_index=None to
 809           # avoid being used in the future. # The current code does not use
 810           # example_index of training data.
 811       if is_training:
 812         feat_example_index = None
 813       else:
 814         feat_example_index = example_index
 815 
 816       feature = InputFeatures(
 817           unique_id=unique_id,
 818           example_index=feat_example_index,
 819           doc_span_index=doc_span_index,
 820           tok_start_to_orig_index=cur_tok_start_to_orig_index,
 821           tok_end_to_orig_index=cur_tok_end_to_orig_index,
 822           token_is_max_context=token_is_max_context,
 823           input_ids=input_ids,
 824           input_mask=input_mask,
 825           p_mask=p_mask,
 826           segment_ids=segment_ids,
 827           paragraph_len=paragraph_len,
 828           cls_index=cls_index,
 829           start_position=start_position,
 830           end_position=end_position,
 831           is_impossible=span_is_impossible)
 832 
 833       # Run callback
 834       output_fn(feature)
 835 
 836       unique_id += 1
 837       if span_is_impossible:
 838         cnt_neg += 1
 839       else:
 840         cnt_pos += 1
 841 
 842   logging.info("Total number of instances: %d = pos %d + neg %d",
 843                cnt_pos + cnt_neg, cnt_pos, cnt_neg)

 846 def _check_is_max_context(doc_spans, cur_span_index, position):
 847   """Check if this is the "max context" doc span for the token."""
 848 
 849   # Because of the sliding window approach taken to scoring documents, a single
 850   # token can appear in multiple documents. E.g.
 851   #  Doc: the man went to the store and bought a gallon of milk
 852   #  Span A: the man went to the
 853   #  Span B: to the store and bought
 854   #  Span C: and bought a gallon of
 855   #  ...
 856   #
 857   # Now the word "bought" will have two scores from spans B and C. We only
 858   # want to consider the score with "maximum context", which we define as
 859   # the *minimum* of its left and right context (the *sum* of left and
 860   # right context will always be the same, of course).
 861   #
 862   # In the example the maximum context for "bought" would be span C since
 863   # it has 1 left context and 3 right context, while span B has 4 left context
 864   # and 0 right context.
 865   best_score = None
 866   best_span_index = None
 867   for (span_index, doc_span) in enumerate(doc_spans):
 868     end = doc_span.start + doc_span.length - 1
 869     if position < doc_span.start:
 870       continue
 871     if position > end:
 872       continue
 873     num_left_context = position - doc_span.start
 874     num_right_context = end - position
 875     score = min(num_left_context, num_right_context) + 0.01 * doc_span.length
 876     if best_score is None or score > best_score:
 877       best_score = score
 878       best_span_index = span_index
 879 
 880   return cur_span_index == best_span_index

 883 class FeatureWriter(object):
 884   """Writes InputFeature to TF example file."""
 885 
 886   def __init__(self, filename, is_training):
 887     self.filename = filename
 888     self.is_training = is_training
 889     self.num_features = 0
 890     self._writer = tf.io.TFRecordWriter(filename)
 891 
 892   def process_feature(self, feature):
 893     """Write a InputFeature to the TFRecordWriter as a tf.train.Example."""
 894     self.num_features += 1
 895 
 896     def create_int_feature(values):
 897       feature = tf.train.Feature(
 898           int64_list=tf.train.Int64List(value=list(values)))
 899       return feature
 900 
 901     def create_float_feature(values):
 902       f = tf.train.Feature(float_list=tf.train.FloatList(value=list(values)))
 903       return f
 904 
 905     features = collections.OrderedDict()
 906     features["unique_ids"] = create_int_feature([feature.unique_id])
 907     features["input_ids"] = create_int_feature(feature.input_ids)
 908     features["input_mask"] = create_float_feature(feature.input_mask)
 909     features["p_mask"] = create_float_feature(feature.p_mask)
 910     features["segment_ids"] = create_int_feature(feature.segment_ids)
 911 
 912     features["cls_index"] = create_int_feature([feature.cls_index])
 913 
 914     if self.is_training:
 915       features["start_positions"] = create_int_feature([feature.start_position])
 916       features["end_positions"] = create_int_feature([feature.end_position])
 917       impossible = 0
 918       if feature.is_impossible:
 919         impossible = 1
 920       features["is_impossible"] = create_float_feature([impossible])
 921 
 922     tf_example = tf.train.Example(features=tf.train.Features(feature=features))
 923     self._writer.write(tf_example.SerializeToString())
 924 
 925   def close(self):
 926     self._writer.close()
