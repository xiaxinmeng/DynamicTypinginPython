  31 def inception_v2_base(inputs,
  32                       final_endpoint='Mixed_5c',
  33                       min_depth=16,
  34                       depth_multiplier=1.0,
  35                       use_separable_conv=True,
  36                       data_format='NHWC',
  37                       include_root_block=True,
  38                       scope=None):
  39   """Inception v2 (6a2).
  40 
  41   Constructs an Inception v2 network from inputs to the given final endpoint.
  42   This method can construct the network up to the layer inception(5b) as
  43   described in http://arxiv.org/abs/1502.03167.
  44 
  45   Args:
  46     inputs: a tensor of shape [batch_size, height, width, channels].
  47     final_endpoint: specifies the endpoint to construct the network up to. It
  48       can be one of ['Conv2d_1a_7x7', 'MaxPool_2a_3x3', 'Conv2d_2b_1x1',
  49       'Conv2d_2c_3x3', 'MaxPool_3a_3x3', 'Mixed_3b', 'Mixed_3c', 'Mixed_4a',
  50       'Mixed_4b', 'Mixed_4c', 'Mixed_4d', 'Mixed_4e', 'Mixed_5a', 'Mixed_5b',
  51       'Mixed_5c']. If include_root_block is False, ['Conv2d_1a_7x7',
  52       'MaxPool_2a_3x3', 'Conv2d_2b_1x1', 'Conv2d_2c_3x3', 'MaxPool_3a_3x3'] will
  53       not be available.
  54     min_depth: Minimum depth value (number of channels) for all convolution ops.
  55       Enforced when depth_multiplier < 1, and not an active constraint when
  56       depth_multiplier >= 1.
  57     depth_multiplier: Float multiplier for the depth (number of channels)
  58       for all convolution ops. The value must be greater than zero. Typical
  59       usage will be to set this value in (0, 1) to reduce the number of
  60       parameters or computation cost of the model.
  61     use_separable_conv: Use a separable convolution for the first layer
  62       Conv2d_1a_7x7. If this is False, use a normal convolution instead.
  63     data_format: Data format of the activations ('NHWC' or 'NCHW').
  64     include_root_block: If True, include the convolution and max-pooling layers
  65       before the inception modules. If False, excludes those layers.
  66     scope: Optional variable_scope.
  67 
  68   Returns:
  69     tensor_out: output tensor corresponding to the final_endpoint.
  70     end_points: a set of activations for external use, for example summaries or
  71                 losses.
  72 
  73   Raises:
  74     ValueError: if final_endpoint is not set to one of the predefined values,
  75                 or depth_multiplier <= 0
  76   """
  77 
  78   # end_points will collect relevant activations for external use, for example
  79   # summaries or losses.
  80   end_points = {}
  81 
  82   # Used to find thinned depths for each layer.
  83   if depth_multiplier <= 0:
  84     raise ValueError('depth_multiplier is not greater than zero.')
  85   depth = lambda d: max(int(d * depth_multiplier), min_depth)
  86 
  87   if data_format != 'NHWC' and data_format != 'NCHW':
  88     raise ValueError('data_format must be either NHWC or NCHW.')
  89   if data_format == 'NCHW' and use_separable_conv:
  90     raise ValueError(
  91         'separable convolution only supports NHWC layout. NCHW data format can'
  92         ' only be used when use_separable_conv is False.'
  93     )
  94 
  95   concat_dim = 3 if data_format == 'NHWC' else 1
  96   with tf.variable_scope(scope, 'InceptionV2', [inputs]):
  97     with slim.arg_scope(
  98         [slim.conv2d, slim.max_pool2d, slim.avg_pool2d],
  99         stride=1,
 100         padding='SAME',
 101         data_format=data_format):
 102 
 103       net = inputs
 104       if include_root_block:
 105         # Note that sizes in the comments below assume an input spatial size of
 106         # 224x224, however, the inputs can be of any size greater 32x32.
 107 
 108         # 224 x 224 x 3
 109         end_point = 'Conv2d_1a_7x7'
 110 
 111         if use_separable_conv:
 112           # depthwise_multiplier here is different from depth_multiplier.
 113           # depthwise_multiplier determines the output channels of the initial
 114           # depthwise conv (see docs for tf.nn.separable_conv2d), while
 115           # depth_multiplier controls the # channels of the subsequent 1x1
 116           # convolution. Must have
 117           #   in_channels * depthwise_multipler <= out_channels
 118           # so that the separable convolution is not overparameterized.
 119           depthwise_multiplier = min(int(depth(64) / 3), 8)
 120           net = slim.separable_conv2d(
 121               inputs,
 122               depth(64), [7, 7],
 123               depth_multiplier=depthwise_multiplier,
 124               stride=2,
 125               padding='SAME',
 126               weights_initializer=trunc_normal(1.0),
 127               scope=end_point)
 128         else:
 129           # Use a normal convolution instead of a separable convolution.
 130           net = slim.conv2d(
 131               inputs,
 132               depth(64), [7, 7],
 133               stride=2,
 134               weights_initializer=trunc_normal(1.0),
 135               scope=end_point)
 136         end_points[end_point] = net
 137         if end_point == final_endpoint:
 138           return net, end_points
 139         # 112 x 112 x 64
 140         end_point = 'MaxPool_2a_3x3'
 141         net = slim.max_pool2d(net, [3, 3], scope=end_point, stride=2)
 142         end_points[end_point] = net
 143         if end_point == final_endpoint:
 144           return net, end_points
 145         # 56 x 56 x 64
 146         end_point = 'Conv2d_2b_1x1'
 147         net = slim.conv2d(
 148             net,
 149             depth(64), [1, 1],
 150             scope=end_point,
 151             weights_initializer=trunc_normal(0.1))
 152         end_points[end_point] = net
 153         if end_point == final_endpoint:
 154           return net, end_points
 155         # 56 x 56 x 64
 156         end_point = 'Conv2d_2c_3x3'
 157         net = slim.conv2d(net, depth(192), [3, 3], scope=end_point)
 158         end_points[end_point] = net
 159         if end_point == final_endpoint:
 160           return net, end_points
 161         # 56 x 56 x 192
 162         end_point = 'MaxPool_3a_3x3'
 163         net = slim.max_pool2d(net, [3, 3], scope=end_point, stride=2)
 164         end_points[end_point] = net
 165         if end_point == final_endpoint:
 166           return net, end_points
 167 
 168       # 28 x 28 x 192
 169       # Inception module.
 170       end_point = 'Mixed_3b'
 171       with tf.variable_scope(end_point):
 172         with tf.variable_scope('Branch_0'):
 173           branch_0 = slim.conv2d(net, depth(64), [1, 1], scope='Conv2d_0a_1x1')
 174         with tf.variable_scope('Branch_1'):
 175           branch_1 = slim.conv2d(
 176               net, depth(64), [1, 1],
 177               weights_initializer=trunc_normal(0.09),
 178               scope='Conv2d_0a_1x1')
 179           branch_1 = slim.conv2d(branch_1, depth(64), [3, 3],
 180                                  scope='Conv2d_0b_3x3')
 181         with tf.variable_scope('Branch_2'):
 182           branch_2 = slim.conv2d(
 183               net, depth(64), [1, 1],
 184               weights_initializer=trunc_normal(0.09),
 185               scope='Conv2d_0a_1x1')
 186           branch_2 = slim.conv2d(branch_2, depth(96), [3, 3],
 187                                  scope='Conv2d_0b_3x3')
 188           branch_2 = slim.conv2d(branch_2, depth(96), [3, 3],
 189                                  scope='Conv2d_0c_3x3')
 190         with tf.variable_scope('Branch_3'):
 191           branch_3 = slim.avg_pool2d(net, [3, 3], scope='AvgPool_0a_3x3')
 192           branch_3 = slim.conv2d(
 193               branch_3, depth(32), [1, 1],
 194               weights_initializer=trunc_normal(0.1),
 195               scope='Conv2d_0b_1x1')
 196         net = tf.concat(
 197             axis=concat_dim, values=[branch_0, branch_1, branch_2, branch_3])
 198         end_points[end_point] = net
 199         if end_point == final_endpoint: return net, end_points
 200       # 28 x 28 x 256
 201       end_point = 'Mixed_3c'
 202       with tf.variable_scope(end_point):
 203         with tf.variable_scope('Branch_0'):
 204           branch_0 = slim.conv2d(net, depth(64), [1, 1], scope='Conv2d_0a_1x1')
 205         with tf.variable_scope('Branch_1'):
 206           branch_1 = slim.conv2d(
 207               net, depth(64), [1, 1],
 208               weights_initializer=trunc_normal(0.09),
 209               scope='Conv2d_0a_1x1')
 210           branch_1 = slim.conv2d(branch_1, depth(96), [3, 3],
 211                                  scope='Conv2d_0b_3x3')
 212         with tf.variable_scope('Branch_2'):
 213           branch_2 = slim.conv2d(
 214               net, depth(64), [1, 1],
 215               weights_initializer=trunc_normal(0.09),
 216               scope='Conv2d_0a_1x1')
 217           branch_2 = slim.conv2d(branch_2, depth(96), [3, 3],
 218                                  scope='Conv2d_0b_3x3')
 219           branch_2 = slim.conv2d(branch_2, depth(96), [3, 3],
 220                                  scope='Conv2d_0c_3x3')
 221         with tf.variable_scope('Branch_3'):
 222           branch_3 = slim.avg_pool2d(net, [3, 3], scope='AvgPool_0a_3x3')
 223           branch_3 = slim.conv2d(
 224               branch_3, depth(64), [1, 1],
 225               weights_initializer=trunc_normal(0.1),
 226               scope='Conv2d_0b_1x1')
 227         net = tf.concat(
 228             axis=concat_dim, values=[branch_0, branch_1, branch_2, branch_3])
 229         end_points[end_point] = net
 230         if end_point == final_endpoint: return net, end_points
 231       # 28 x 28 x 320
 232       end_point = 'Mixed_4a'
 233       with tf.variable_scope(end_point):
 234         with tf.variable_scope('Branch_0'):
 235           branch_0 = slim.conv2d(
 236               net, depth(128), [1, 1],
 237               weights_initializer=trunc_normal(0.09),
 238               scope='Conv2d_0a_1x1')
 239           branch_0 = slim.conv2d(branch_0, depth(160), [3, 3], stride=2,
 240                                  scope='Conv2d_1a_3x3')
 241         with tf.variable_scope('Branch_1'):
 242           branch_1 = slim.conv2d(
 243               net, depth(64), [1, 1],
 244               weights_initializer=trunc_normal(0.09),
 245               scope='Conv2d_0a_1x1')
 246           branch_1 = slim.conv2d(
 247               branch_1, depth(96), [3, 3], scope='Conv2d_0b_3x3')
 248           branch_1 = slim.conv2d(
 249               branch_1, depth(96), [3, 3], stride=2, scope='Conv2d_1a_3x3')
 250         with tf.variable_scope('Branch_2'):
 251           branch_2 = slim.max_pool2d(
 252               net, [3, 3], stride=2, scope='MaxPool_1a_3x3')
 253         net = tf.concat(axis=concat_dim, values=[branch_0, branch_1, branch_2])
 254         end_points[end_point] = net
 255         if end_point == final_endpoint: return net, end_points
 256       # 14 x 14 x 576
 257       end_point = 'Mixed_4b'
 258       with tf.variable_scope(end_point):
 259         with tf.variable_scope('Branch_0'):
 260           branch_0 = slim.conv2d(net, depth(224), [1, 1], scope='Conv2d_0a_1x1')
 261         with tf.variable_scope('Branch_1'):
 262           branch_1 = slim.conv2d(
 263               net, depth(64), [1, 1],
 264               weights_initializer=trunc_normal(0.09),
 265               scope='Conv2d_0a_1x1')
 266           branch_1 = slim.conv2d(
 267               branch_1, depth(96), [3, 3], scope='Conv2d_0b_3x3')
 268         with tf.variable_scope('Branch_2'):
 269           branch_2 = slim.conv2d(
 270               net, depth(96), [1, 1],
 271               weights_initializer=trunc_normal(0.09),
 272               scope='Conv2d_0a_1x1')
 273           branch_2 = slim.conv2d(branch_2, depth(128), [3, 3],
 274                                  scope='Conv2d_0b_3x3')
 275           branch_2 = slim.conv2d(branch_2, depth(128), [3, 3],
 276                                  scope='Conv2d_0c_3x3')
 277         with tf.variable_scope('Branch_3'):
 278           branch_3 = slim.avg_pool2d(net, [3, 3], scope='AvgPool_0a_3x3')
 279           branch_3 = slim.conv2d(
 280               branch_3, depth(128), [1, 1],
 281               weights_initializer=trunc_normal(0.1),
 282               scope='Conv2d_0b_1x1')
 283         net = tf.concat(
 284             axis=concat_dim, values=[branch_0, branch_1, branch_2, branch_3])
 285         end_points[end_point] = net
 286         if end_point == final_endpoint: return net, end_points
 287       # 14 x 14 x 576
 288       end_point = 'Mixed_4c'
 289       with tf.variable_scope(end_point):
 290         with tf.variable_scope('Branch_0'):
 291           branch_0 = slim.conv2d(net, depth(192), [1, 1], scope='Conv2d_0a_1x1')
 292         with tf.variable_scope('Branch_1'):
 293           branch_1 = slim.conv2d(
 294               net, depth(96), [1, 1],
 295               weights_initializer=trunc_normal(0.09),
 296               scope='Conv2d_0a_1x1')
 297           branch_1 = slim.conv2d(branch_1, depth(128), [3, 3],
 298                                  scope='Conv2d_0b_3x3')
 299         with tf.variable_scope('Branch_2'):
 300           branch_2 = slim.conv2d(
 301               net, depth(96), [1, 1],
 302               weights_initializer=trunc_normal(0.09),
 303               scope='Conv2d_0a_1x1')
 304           branch_2 = slim.conv2d(branch_2, depth(128), [3, 3],
 305                                  scope='Conv2d_0b_3x3')
 306           branch_2 = slim.conv2d(branch_2, depth(128), [3, 3],
 307                                  scope='Conv2d_0c_3x3')
 308         with tf.variable_scope('Branch_3'):
 309           branch_3 = slim.avg_pool2d(net, [3, 3], scope='AvgPool_0a_3x3')
 310           branch_3 = slim.conv2d(
 311               branch_3, depth(128), [1, 1],
 312               weights_initializer=trunc_normal(0.1),
 313               scope='Conv2d_0b_1x1')
 314         net = tf.concat(
 315             axis=concat_dim, values=[branch_0, branch_1, branch_2, branch_3])
 316         end_points[end_point] = net
 317         if end_point == final_endpoint: return net, end_points
 318       # 14 x 14 x 576
 319       end_point = 'Mixed_4d'
 320       with tf.variable_scope(end_point):
 321         with tf.variable_scope('Branch_0'):
 322           branch_0 = slim.conv2d(net, depth(160), [1, 1], scope='Conv2d_0a_1x1')
 323         with tf.variable_scope('Branch_1'):
 324           branch_1 = slim.conv2d(
 325               net, depth(128), [1, 1],
 326               weights_initializer=trunc_normal(0.09),
 327               scope='Conv2d_0a_1x1')
 328           branch_1 = slim.conv2d(branch_1, depth(160), [3, 3],
 329                                  scope='Conv2d_0b_3x3')
 330         with tf.variable_scope('Branch_2'):
 331           branch_2 = slim.conv2d(
 332               net, depth(128), [1, 1],
 333               weights_initializer=trunc_normal(0.09),
 334               scope='Conv2d_0a_1x1')
 335           branch_2 = slim.conv2d(branch_2, depth(160), [3, 3],
 336                                  scope='Conv2d_0b_3x3')
 337           branch_2 = slim.conv2d(branch_2, depth(160), [3, 3],
 338                                  scope='Conv2d_0c_3x3')
 339         with tf.variable_scope('Branch_3'):
 340           branch_3 = slim.avg_pool2d(net, [3, 3], scope='AvgPool_0a_3x3')
 341           branch_3 = slim.conv2d(
 342               branch_3, depth(96), [1, 1],
 343               weights_initializer=trunc_normal(0.1),
 344               scope='Conv2d_0b_1x1')
 345         net = tf.concat(
 346             axis=concat_dim, values=[branch_0, branch_1, branch_2, branch_3])
 347         end_points[end_point] = net
 348         if end_point == final_endpoint: return net, end_points
 349       # 14 x 14 x 576
 350       end_point = 'Mixed_4e'
 351       with tf.variable_scope(end_point):
 352         with tf.variable_scope('Branch_0'):
 353           branch_0 = slim.conv2d(net, depth(96), [1, 1], scope='Conv2d_0a_1x1')
 354         with tf.variable_scope('Branch_1'):
 355           branch_1 = slim.conv2d(
 356               net, depth(128), [1, 1],
 357               weights_initializer=trunc_normal(0.09),
 358               scope='Conv2d_0a_1x1')
 359           branch_1 = slim.conv2d(branch_1, depth(192), [3, 3],
 360                                  scope='Conv2d_0b_3x3')
 361         with tf.variable_scope('Branch_2'):
 362           branch_2 = slim.conv2d(
 363               net, depth(160), [1, 1],
 364               weights_initializer=trunc_normal(0.09),
 365               scope='Conv2d_0a_1x1')
 366           branch_2 = slim.conv2d(branch_2, depth(192), [3, 3],
 367                                  scope='Conv2d_0b_3x3')
 368           branch_2 = slim.conv2d(branch_2, depth(192), [3, 3],
 369                                  scope='Conv2d_0c_3x3')

 372           branch_3 = slim.conv2d(
 373               branch_3, depth(96), [1, 1],
 374               weights_initializer=trunc_normal(0.1),

 392               net, depth(192), [1, 1],

 400           branch_2 = slim.max_pool2d(net, [3, 3], stride=2,

 417                                  scope='Conv2d_0b_3x3')

 435         end_points[end_point] = net

 437       # 7 x 7 x 1024

 471 def inception_v2(inputs,
 472                  num_classes=1000,
 473                  is_training=True,
 474                  dropout_keep_prob=0.8,
 475                  min_depth=16,
 476                  depth_multiplier=1.0,
 477                  prediction_fn=slim.softmax,
 478                  spatial_squeeze=True,
 479                  reuse=None,
 480                  scope='InceptionV2',
 481                  global_pool=False):
 482   """Inception v2 model for classification.
 483 
 484   Constructs an Inception v2 network for classification as described in
 485   http://arxiv.org/abs/1502.03167.
 486 
 487   The default image size used to train this network is 224x224.
 488 
 489   Args:
 490     inputs: a tensor of shape [batch_size, height, width, channels].
 491     num_classes: number of predicted classes. If 0 or None, the logits layer
 492       is omitted and the input features to the logits layer (before dropout)
 493       are returned instead.
 494     is_training: whether is training or not.
 495     dropout_keep_prob: the percentage of activation values that are retained.
 496     min_depth: Minimum depth value (number of channels) for all convolution ops.
 497       Enforced when depth_multiplier < 1, and not an active constraint when
 498       depth_multiplier >= 1.
 499     depth_multiplier: Float multiplier for the depth (number of channels)
 500       for all convolution ops. The value must be greater than zero. Typical
 501       usage will be to set this value in (0, 1) to reduce the number of
 502       parameters or computation cost of the model.
 503     prediction_fn: a function to get predictions out of logits.
 504     spatial_squeeze: if True, logits is of shape [B, C], if false logits is of
 505         shape [B, 1, 1, C], where B is batch_size and C is number of classes.
 506     reuse: whether or not the network and its variables should be reused. To be
 507       able to reuse 'scope' must be given.
 508     scope: Optional variable_scope.
 509     global_pool: Optional boolean flag to control the avgpooling before the
 510       logits layer. If false or unset, pooling is done with a fixed window
 511       that reduces default-sized inputs to 1x1, while larger inputs lead to
 512       larger outputs. If true, any input size is pooled down to 1x1.
 513 
 514   Returns:
 515     net: a Tensor with the logits (pre-softmax activations) if num_classes
 516       is a non-zero integer, or the non-dropped-out input to the logits layer
 517       if num_classes is 0 or None.
 518     end_points: a dictionary from components of the network to the corresponding
 519       activation.
 520 
 521   Raises:
 522     ValueError: if final_endpoint is not set to one of the predefined values,
 523                 or depth_multiplier <= 0
 524   """
 525   if depth_multiplier <= 0:
 526     raise ValueError('depth_multiplier is not greater than zero.')
 527 
 528   # Final pooling and prediction
 529   with tf.variable_scope(
 530       scope, 'InceptionV2', [inputs], reuse=reuse) as scope:
 531     with slim.arg_scope([slim.batch_norm, slim.dropout],
 532                         is_training=is_training):
 533       net, end_points = inception_v2_base(
 534           inputs, scope=scope, min_depth=min_depth,
 535           depth_multiplier=depth_multiplier)
 536       with tf.variable_scope('Logits'):
 537         if global_pool:
 538           # Global average pooling.
 539           net = tf.reduce_mean(
 540               input_tensor=net, axis=[1, 2], keepdims=True, name='global_pool')
 541           end_points['global_pool'] = net
 542         else:
 543           # Pooling with a fixed kernel size.
 544           kernel_size = _reduced_kernel_size_for_small_input(net, [7, 7])
 545           net = slim.avg_pool2d(net, kernel_size, padding='VALID',
 546                                 scope='AvgPool_1a_{}x{}'.format(*kernel_size))
 547           end_points['AvgPool_1a'] = net
 548         if not num_classes:
 549           return net, end_points
 550         # 1 x 1 x 1024
 551         net = slim.dropout(net, keep_prob=dropout_keep_prob, scope='Dropout_1b')
 552         end_points['PreLogits'] = net
 553         logits = slim.conv2d(net, num_classes, [1, 1], activation_fn=None,
 554                              normalizer_fn=None, scope='Conv2d_1c_1x1')
 555         if spatial_squeeze:
 556           logits = tf.squeeze(logits, [1, 2], name='SpatialSqueeze')
 557       end_points['Logits'] = logits
 558       end_points['Predictions'] = prediction_fn(logits, scope='Predictions')
 559   return logits, end_points
 560 inception_v2.default_image_size = 224
 561

 563 def _reduced_kernel_size_for_small_input(input_tensor, kernel_size):
 564   """Define kernel size which is automatically reduced for small input.
 565 
 566   If the shape of the input images is unknown at graph construction time this
 567   function assumes that the input images are is large enough.
 568 
 569   Args:
 570     input_tensor: input tensor of size [batch_size, height, width, channels].
 571     kernel_size: desired kernel size of length 2: [kernel_height, kernel_width]
 572 
 573   Returns:
 574     a tensor with the kernel size.
 575 
 576   TODO(jrru): Make this function work with unknown shapes. Theoretically, this
 577   can be done with the code below. Problems are two-fold: (1) If the shape was
 578   known, it will be lost. (2) inception.slim.ops._two_element_tuple cannot
 579   handle tensors that define the kernel size.
 580       shape = tf.shape(input_tensor)
 581       return = tf.stack([tf.minimum(shape[1], kernel_size[0]),
 582                          tf.minimum(shape[2], kernel_size[1])])
 583 
 584   """
 585   shape = input_tensor.get_shape().as_list()
 586   if shape[1] is None or shape[2] is None:
 587     kernel_size_out = kernel_size
 588   else:
 589     kernel_size_out = [min(shape[1], kernel_size[0]),
 590                        min(shape[2], kernel_size[1])]
 591   return kernel_size_out
